47 """"A Printer to convert python expressions to strings of javascript code

87 def _get_loop_opening_ending(self, indices): 

88 open_lines = [] 

89 close_lines = [] 

90 loopstart = "for (var %(varble)s=%(start)s; %(varble)s<%(end)s; %(varble)s++){" 

91 for i in indices: 

92 # Javascript arrays start at 0 and end at dimension-1 

93 open_lines.append(loopstart % { 

94 'varble': self._print(i.label), 

95 'start': self._print(i.lower), 

96 'end': self._print(i.upper + 1)}) 

97 close_lines.append("}") 

98 return open_lines, close_lines

100 def _print_Pow(self, expr): 

101 PREC = precedence(expr) 

102 if expr.exp == -1: 

103 return '1/%s' % (self.parenthesize(expr.base, PREC)) 

104 elif expr.exp == 0.5: 

105 return 'Math.sqrt(%s)' % self._print(expr.base) 

106 elif expr.exp == S(1)/3: 

107 return 'Math.cbrt(%s)' % self._print(expr.base) 

108 else: 

109 return 'Math.pow(%s, %s)' % (self._print(expr.base), 

110 self._print(expr.exp))

112 def _print_Rational(self, expr): 

113 p, q = int(expr.p), int(expr.q) 

114 return '%d/%d' % (p, q)

116 def _print_Indexed(self, expr): 

117 # calculate index for 1d array 

118 dims = expr.shape 

119 elem = S.Zero 

120 offset = S.One 

121 for i in reversed(range(expr.rank)): 

122 elem += expr.indices[i]*offset 

123 offset *= dims[i] 

124 return "%s[%s]" % (self._print(expr.base.label), self._print(elem))

141 def _print_Piecewise(self, expr): 

142 if expr.args[-1].cond != True: 

143 # We need the last conditional to be a True, otherwise the resulting 

144 # function may not return a result. 

145 raise ValueError("All Piecewise expressions must contain an " 

146 "(expr, True) statement to be used as a default " 

147 "condition. Without one, the generated " 

148 "expression may not evaluate to anything under " 

149 "some condition.") 

150 lines = [] 

151 if expr.has(Assignment): 

152 for i, (e, c) in enumerate(expr.args): 

153 if i == 0: 

154 lines.append("if (%s) {" % self._print(c)) 

155 elif i == len(expr.args) - 1 and c == True: 

156 lines.append("else {") 

157 else: 

158 lines.append("else if (%s) {" % self._print(c)) 

159 code0 = self._print(e) 

160 lines.append(code0) 

161 lines.append("}") 

162 return "\n".join(lines) 

163 else: 

164 # The piecewise was used in an expression, need to do inline 

165 # operators. This has the downside that inline operators will 

166 # not work for statements that span multiple lines (Matrix or 

167 # Indexed expressions). 

168 ecpairs = ["((%s) ? (\n%s\n)\n" % (self._print(c), self._print(e)) 

169 for e, c in expr.args[:-1]] 

170 last_line = ": (\n%s\n)" % self._print(expr.args[-1].expr) 

171 return ": ".join(ecpairs) + last_line + " ".join([")"*len(ecpairs)])

174 return "{0}[{1}]".format(self.parenthesize(expr.parent, 

175 PRECEDENCE["Atom"], strict=True), 

176 expr.j + expr.i*expr.parent.shape[1])

178 def indent_code(self, code): 

179 """Accepts a string of code or a list of code lines""" 

180 

181 if isinstance(code, string_types): 

182 code_lines = self.indent_code(code.splitlines(True)) 

183 return ''.join(code_lines) 

184 

185 tab = " " 

186 inc_token = ('{', '(', '{\n', '(\n') 

187 dec_token = ('}', ')') 

188 

189 code = [ line.lstrip(' \t') for line in code ] 

190 

191 increase = [ int(any(map(line.endswith, inc_token))) for line in code ] 

192 decrease = [ int(any(map(line.startswith, dec_token))) 

193 for line in code ] 

194 

195 pretty = [] 

196 level = 0 

197 for n, line in enumerate(code): 

198 if line == '' or line == '\n': 

199 pretty.append(line) 

200 continue 

201 level -= decrease[n] 

202 pretty.append("%s%s" % (tab*level, line)) 

203 level += increase[n] 

204 return pretty

207def jscode(expr, assign_to=None, **settings): 

208 """Converts an expr to a string of javascript code 

209 

210 Parameters 

211 ========== 

212 

213 expr : Expr 

214 A sympy expression to be converted. 

215 assign_to : optional 

216 When given, the argument is used as the name of the variable to which 

217 the expression is assigned. Can be a string, ``Symbol``, 

218 ``MatrixSymbol``, or ``Indexed`` type. This is helpful in case of 

219 line-wrapping, or for expressions that generate multi-line statements. 

220 precision : integer, optional 

221 The precision for numbers such as pi [default=15]. 

222 user_functions : dict, optional 

223 A dictionary where keys are ``FunctionClass`` instances and values are 

224 their string representations. Alternatively, the dictionary value can 

225 be a list of tuples i.e. [(argument_test, js_function_string)]. See 

226 below for examples. 

227 human : bool, optional 

228 If True, the result is a single string that may contain some constant 

229 declarations for the number symbols. If False, the same information is 

230 returned in a tuple of (symbols_to_declare, not_supported_functions, 

231 code_text). [default=True]. 

232 contract: bool, optional 

233 If True, ``Indexed`` instances are assumed to obey tensor contraction 

234 rules and the corresponding nested loops over indices are generated. 

235 Setting contract=False will not generate loops, instead the user is 

236 responsible to provide values for the indices in the code. 

237 [default=True]. 

238 

239 Examples 

240 ======== 

241 

242 >>> from sympy import jscode, symbols, Rational, sin, ceiling, Abs 

243 >>> x, tau = symbols("x, tau") 

244 >>> jscode((2*tau)**Rational(7, 2)) 

245 '8*Math.sqrt(2)*Math.pow(tau, 7/2)' 

246 >>> jscode(sin(x), assign_to="s") 

247 's = Math.sin(x);' 

248 

249 Custom printing can be defined for certain types by passing a dictionary of 

250 "type" : "function" to the ``user_functions`` kwarg. Alternatively, the 

251 dictionary value can be a list of tuples i.e. [(argument_test, 

252 js_function_string)]. 

253 

254 >>> custom_functions = { 

255 ... "ceiling": "CEIL", 

256 ... "Abs": [(lambda x: not x.is_integer, "fabs"), 

257 ... (lambda x: x.is_integer, "ABS")] 

258 ... } 

259 >>> jscode(Abs(x) + ceiling(x), user_functions=custom_functions) 

260 'fabs(x) + CEIL(x)' 

261 

262 ``Piecewise`` expressions are converted into conditionals. If an 

263 ``assign_to`` variable is provided an if statement is created, otherwise 

264 the ternary operator is used. Note that if the ``Piecewise`` lacks a 

265 default term, represented by ``(expr, True)`` then an error will be thrown. 

266 This is to prevent generating an expression that may not evaluate to 

267 anything. 

268 

269 >>> from sympy import Piecewise 

270 >>> expr = Piecewise((x + 1, x > 0), (x, True)) 

271 >>> print(jscode(expr, tau)) 

272 if (x > 0) { 

273 tau = x + 1; 

274 } 

275 else { 

276 tau = x; 

277 } 

278 

279 Support for loops is provided through ``Indexed`` types. With 

280 ``contract=True`` these expressions will be turned into loops, whereas 

281 ``contract=False`` will just print the assignment expression that should be 

282 looped over: 

283 

284 >>> from sympy import Eq, IndexedBase, Idx 

285 >>> len_y = 5 

286 >>> y = IndexedBase('y', shape=(len_y,)) 

287 >>> t = IndexedBase('t', shape=(len_y,)) 

288 >>> Dy = IndexedBase('Dy', shape=(len_y-1,)) 

289 >>> i = Idx('i', len_y-1) 

290 >>> e=Eq(Dy[i], (y[i+1]-y[i])/(t[i+1]-t[i])) 

291 >>> jscode(e.rhs, assign_to=e.lhs, contract=False) 

292 'Dy[i] = (y[i + 1] - y[i])/(t[i + 1] - t[i]);' 

293 

294 Matrices are also supported, but a ``MatrixSymbol`` of the same dimensions 

295 must be provided to ``assign_to``. Note that any expression that can be 

296 generated normally can also exist inside a Matrix: 

297 

298 >>> from sympy import Matrix, MatrixSymbol 

299 >>> mat = Matrix([x**2, Piecewise((x + 1, x > 0), (x, True)), sin(x)]) 

300 >>> A = MatrixSymbol('A', 3, 1) 

301 >>> print(jscode(mat, A)) 

302 A[0] = Math.pow(x, 2); 

303 if (x > 0) { 

304 A[1] = x + 1; 

305 } 

306 else { 

307 A[1] = x; 

308 } 

309 A[2] = Math.sin(x); 

310 """ 

311 

312 return JavascriptCodePrinter(settings).doprint(expr, assign_to)
