8def matrix_fglm(F, ring, O_to): 

9 """ 

10 Converts the reduced Groebner basis ``F`` of a zero-dimensional 

11 ideal w.r.t. ``O_from`` to a reduced Groebner basis 

12 w.r.t. ``O_to``. 

13 

14 References 

15 ========== 

16 

17 .. [1] J.C. Faugere, P. Gianni, D. Lazard, T. Mora (1994). Efficient 

18 Computation of Zero-dimensional Groebner Bases by Change of 

19 Ordering 

20 """ 

21 domain = ring.domain 

22 ngens = ring.ngens 

23 

24 ring_to = ring.clone(order=O_to) 

25 

26 old_basis = _basis(F, ring) 

27 M = _representing_matrices(old_basis, F, ring) 

28 

29 # V contains the normalforms (wrt O_from) of S 

30 S = [ring.zero_monom] 

31 V = [[domain.one] + [domain.zero] * (len(old_basis) - 1)] 

32 G = [] 

33 

34 L = [(i, 0) for i in range(ngens)] # (i, j) corresponds to x_i * S[j] 

35 L.sort(key=lambda k_l: O_to(_incr_k(S[k_l[1]], k_l[0])), reverse=True) 

36 t = L.pop() 

37 

38 P = _identity_matrix(len(old_basis), domain) 

39 

40 while True: 

41 s = len(S) 

42 v = _matrix_mul(M[t[0]], V[t[1]]) 

43 _lambda = _matrix_mul(P, v) 

44 

45 if all(_lambda[i] == domain.zero for i in range(s, len(old_basis))): 

46 # there is a linear combination of v by V 

47 lt = ring.term_new(_incr_k(S[t[1]], t[0]), domain.one) 

48 rest = ring.from_dict({S[i]: _lambda[i] for i in range(s)}) 

49 

50 g = (lt - rest).set_ring(ring_to) 

51 if g: 

52 G.append(g) 

53 else: 

54 # v is linearly independent from V 

55 P = _update(s, _lambda, P) 

56 S.append(_incr_k(S[t[1]], t[0])) 

57 V.append(v) 

58 

59 L.extend([(i, s) for i in range(ngens)]) 

60 L = list(set(L)) 

61 L.sort(key=lambda k_l: O_to(_incr_k(S[k_l[1]], k_l[0])), reverse=True) 

62 

63 L = [(k, l) for (k, l) in L if all(monomial_div(_incr_k(S[l], k), g.LM) is None for g in G)] 

64 

65 if not L: 

66 G = [ g.monic() for g in G ] 

67 return sorted(G, key=lambda g: O_to(g.LM), reverse=True) 

68 

69 t = L.pop()

72def _incr_k(m, k): 

73 return tuple(list(m[:k]) + [m[k] + 1] + list(m[k + 1:]))

76def _identity_matrix(n, domain): 

77 M = [[domain.zero]*n for _ in range(n)] 

78 

79 for i in range(n): 

80 M[i][i] = domain.one 

81 

82 return M

85def _matrix_mul(M, v): 

86 return [sum([row[i] * v[i] for i in range(len(v))]) for row in M]

89def _update(s, _lambda, P): 

90 """ 

91 Update ``P`` such that for the updated `P'` `P' v = e_{s}`. 

92 """ 

93 k = min([j for j in range(s, len(_lambda)) if _lambda[j] != 0]) 

94 

95 for r in range(len(_lambda)): 

96 if r != k: 

97 P[r] = [P[r][j] - (P[k][j] * _lambda[r]) / _lambda[k] for j in range(len(P[r]))] 

98 

99 P[k] = [P[k][j] / _lambda[k] for j in range(len(P[k]))] 

100 P[k], P[s] = P[s], P[k] 

101 

102 return P

105def _representing_matrices(basis, G, ring): 

106 r""" 

107 Compute the matrices corresponding to the linear maps `m \mapsto 

108 x_i m` for all variables `x_i`. 

109 """ 

110 domain = ring.domain 

111 u = ring.ngens-1 

112 

113 def var(i): 

114 return tuple([0] * i + [1] + [0] * (u - i)) 

115 

116 def representing_matrix(m): 

117 M = [[domain.zero] * len(basis) for _ in range(len(basis))] 

118 

119 for i, v in enumerate(basis): 

120 r = ring.term_new(monomial_mul(m, v), domain.one).rem(G) 

121 

122 for monom, coeff in r.terms(): 

123 j = basis.index(monom) 

124 M[j][i] = coeff 

125 

126 return M 

127 

128 return [representing_matrix(var(i)) for i in range(u + 1)]

131def _basis(G, ring): 

132 r""" 

133 Computes a list of monomials which are not divisible by the leading 

134 monomials wrt to ``O`` of ``G``. These monomials are a basis of 

135 `K[X_1, \ldots, X_n]/(G)`. 

136 """ 

137 order = ring.order 

138 

139 leading_monomials = [g.LM for g in G] 

140 candidates = [ring.zero_monom] 

141 basis = [] 

142 

143 while candidates: 

144 t = candidates.pop() 

145 basis.append(t) 

146 

147 new_candidates = [_incr_k(t, k) for k in range(ring.ngens) 

148 if all(monomial_div(_incr_k(t, k), lmg) is None 

149 for lmg in leading_monomials)] 

150 candidates.extend(new_candidates) 

151 candidates.sort(key=lambda m: order(m), reverse=True) 

152 

153 basis = list(set(basis)) 

154 

155 return sorted(basis, key=lambda m: order(m))
