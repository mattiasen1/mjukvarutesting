284 tab = " " 

285 inc_token = ('{', '(', '{\n', '(\n') 

286 dec_token = ('}', ')')

288 code = [ line.lstrip(' \t') for line in code ] 

289 

290 increase = [ int(any(map(line.endswith, inc_token))) for line in code ] 

291 decrease = [ int(any(map(line.startswith, dec_token))) 

292 for line in code ] 

293 

294 pretty = [] 

295 level = 0 

296 for n, line in enumerate(code): 

297 if line == '' or line == '\n': 

298 pretty.append(line) 

299 continue 

300 level -= decrease[n] 

301 pretty.append("%s%s" % (tab*level, line)) 

302 level += increase[n] 

303 return pretty

306def rcode(expr, assign_to=None, **settings): 

307 """Converts an expr to a string of r code 

308 

309 Parameters 

310 ========== 

311 

312 expr : Expr 

313 A sympy expression to be converted. 

314 assign_to : optional 

315 When given, the argument is used as the name of the variable to which 

316 the expression is assigned. Can be a string, ``Symbol``, 

317 ``MatrixSymbol``, or ``Indexed`` type. This is helpful in case of 

318 line-wrapping, or for expressions that generate multi-line statements. 

319 precision : integer, optional 

320 The precision for numbers such as pi [default=15]. 

321 user_functions : dict, optional 

322 A dictionary where the keys are string representations of either 

323 ``FunctionClass`` or ``UndefinedFunction`` instances and the values 

324 are their desired R string representations. Alternatively, the 

325 dictionary value can be a list of tuples i.e. [(argument_test, 

326 rfunction_string)] or [(argument_test, rfunction_formater)]. See below 

327 for examples. 

328 human : bool, optional 

329 If True, the result is a single string that may contain some constant 

330 declarations for the number symbols. If False, the same information is 

331 returned in a tuple of (symbols_to_declare, not_supported_functions, 

332 code_text). [default=True]. 

333 contract: bool, optional 

334 If True, ``Indexed`` instances are assumed to obey tensor contraction 

335 rules and the corresponding nested loops over indices are generated. 

336 Setting contract=False will not generate loops, instead the user is 

337 responsible to provide values for the indices in the code. 

338 [default=True]. 

339 

340 Examples 

341 ======== 

342 

343 >>> from sympy import rcode, symbols, Rational, sin, ceiling, Abs, Function 

344 >>> x, tau = symbols("x, tau") 

345 >>> rcode((2*tau)**Rational(7, 2)) 

346 '8*sqrt(2)*tau^(7.0/2.0)' 

347 >>> rcode(sin(x), assign_to="s") 

348 's = sin(x);' 

349 

350 Simple custom printing can be defined for certain types by passing a 

351 dictionary of {"type" : "function"} to the ``user_functions`` kwarg. 

352 Alternatively, the dictionary value can be a list of tuples i.e. 

353 [(argument_test, cfunction_string)]. 

354 

355 >>> custom_functions = { 

356 ... "ceiling": "CEIL", 

357 ... "Abs": [(lambda x: not x.is_integer, "fabs"), 

358 ... (lambda x: x.is_integer, "ABS")], 

359 ... "func": "f" 

360 ... } 

361 >>> func = Function('func') 

362 >>> rcode(func(Abs(x) + ceiling(x)), user_functions=custom_functions) 

363 'f(fabs(x) + CEIL(x))' 

364 

365 or if the R-function takes a subset of the original arguments: 

366 

367 >>> rcode(2**x + 3**x, user_functions={'Pow': [ 

368 ... (lambda b, e: b == 2, lambda b, e: 'exp2(%s)' % e), 

369 ... (lambda b, e: b != 2, 'pow')]}) 

370 'exp2(x) + pow(3, x)' 

371 

372 ``Piecewise`` expressions are converted into conditionals. If an 

373 ``assign_to`` variable is provided an if statement is created, otherwise 

374 the ternary operator is used. Note that if the ``Piecewise`` lacks a 

375 default term, represented by ``(expr, True)`` then an error will be thrown. 

376 This is to prevent generating an expression that may not evaluate to 

377 anything. 

378 

379 >>> from sympy import Piecewise 

380 >>> expr = Piecewise((x + 1, x > 0), (x, True)) 

381 >>> print(rcode(expr, assign_to=tau)) 

382 tau = ifelse(x > 0,x + 1,x); 

383 

384 Support for loops is provided through ``Indexed`` types. With 

385 ``contract=True`` these expressions will be turned into loops, whereas 

386 ``contract=False`` will just print the assignment expression that should be 

387 looped over: 

388 

389 >>> from sympy import Eq, IndexedBase, Idx 

390 >>> len_y = 5 

391 >>> y = IndexedBase('y', shape=(len_y,)) 

392 >>> t = IndexedBase('t', shape=(len_y,)) 

393 >>> Dy = IndexedBase('Dy', shape=(len_y-1,)) 

394 >>> i = Idx('i', len_y-1) 

395 >>> e=Eq(Dy[i], (y[i+1]-y[i])/(t[i+1]-t[i])) 

396 >>> rcode(e.rhs, assign_to=e.lhs, contract=False) 

397 'Dy[i] = (y[i + 1] - y[i])/(t[i + 1] - t[i]);' 

398 

399 Matrices are also supported, but a ``MatrixSymbol`` of the same dimensions 

400 must be provided to ``assign_to``. Note that any expression that can be 

401 generated normally can also exist inside a Matrix: 

402 

403 >>> from sympy import Matrix, MatrixSymbol 

404 >>> mat = Matrix([x**2, Piecewise((x + 1, x > 0), (x, True)), sin(x)]) 

405 >>> A = MatrixSymbol('A', 3, 1) 

406 >>> print(rcode(mat, A)) 

407 A[0] = x^2; 

408 A[1] = ifelse(x > 0,x + 1,x); 

409 A[2] = sin(x); 

410 

411 """ 

412 

413 return RCodePrinter(settings).doprint(expr, assign_to)
