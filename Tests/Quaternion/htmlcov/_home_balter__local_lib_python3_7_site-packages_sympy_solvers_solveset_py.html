54def _masked(f, *atoms): 

55 """Return ``f``, with all objects given by ``atoms`` replaced with 

56 Dummy symbols, ``d``, and the list of replacements, ``(d, e)``, 

57 where ``e`` is an object of type given by ``atoms`` in which 

58 any other instances of atoms have been recursively replaced with 

59 Dummy symbols, too. The tuples are ordered so that if they are 

60 applied in sequence, the origin ``f`` will be restored. 

61 

62 Examples 

63 ======== 

64 

65 >>> from sympy import cos 

66 >>> from sympy.abc import x 

67 >>> from sympy.solvers.solveset import _masked 

68 

69 >>> f = cos(cos(x) + 1) 

70 >>> f, reps = _masked(cos(1 + cos(x)), cos) 

71 >>> f 

72 _a1 

73 >>> reps 

74 [(_a1, cos(_a0 + 1)), (_a0, cos(x))] 

75 >>> for d, e in reps: 

76 ... f = f.xreplace({d: e}) 

77 >>> f 

78 cos(cos(x) + 1) 

79 """ 

80 sym = numbered_symbols('a', cls=Dummy, real=True) 

81 mask = [] 

82 for a in ordered(f.atoms(*atoms)): 

83 for i in mask: 

84 a = a.replace(*i) 

85 mask.append((a, next(sym))) 

86 for i, (o, n) in enumerate(mask): 

87 f = f.replace(o, n) 

88 mask[i] = (n, o) 

89 mask = list(reversed(mask)) 

90 return f, mask

93def _invert(f_x, y, x, domain=S.Complexes): 

94 r""" 

95 Reduce the complex valued equation ``f(x) = y`` to a set of equations 

96 ``{g(x) = h_1(y), g(x) = h_2(y), ..., g(x) = h_n(y) }`` where ``g(x)`` is 

97 a simpler function than ``f(x)``. The return value is a tuple ``(g(x), 

98 set_h)``, where ``g(x)`` is a function of ``x`` and ``set_h`` is 

99 the set of function ``{h_1(y), h_2(y), ..., h_n(y)}``. 

100 Here, ``y`` is not necessarily a symbol. 

101 

102 The ``set_h`` contains the functions, along with the information 

103 about the domain in which they are valid, through set 

104 operations. For instance, if ``y = Abs(x) - n`` is inverted 

105 in the real domain, then ``set_h`` is not simply 

106 `{-n, n}` as the nature of `n` is unknown; rather, it is: 

107 `Intersection([0, oo) {n}) U Intersection((-oo, 0], {-n})` 

108 

109 By default, the complex domain is used which means that inverting even 

110 seemingly simple functions like ``exp(x)`` will give very different 

111 results from those obtained in the real domain. 

112 (In the case of ``exp(x)``, the inversion via ``log`` is multi-valued 

113 in the complex domain, having infinitely many branches.) 

114 

115 If you are working with real values only (or you are not sure which 

116 function to use) you should probably set the domain to 

117 ``S.Reals`` (or use `invert\_real` which does that automatically). 

118 

119 

120 Examples 

121 ======== 

122 

123 >>> from sympy.solvers.solveset import invert_complex, invert_real 

124 >>> from sympy.abc import x, y 

125 >>> from sympy import exp, log 

126 

127 When does exp(x) == y? 

128 

129 >>> invert_complex(exp(x), y, x) 

130 (x, ImageSet(Lambda(_n, I*(2*_n*pi + arg(y)) + log(Abs(y))), Integers)) 

131 >>> invert_real(exp(x), y, x) 

132 (x, Intersection({log(y)}, Reals)) 

133 

134 When does exp(x) == 1? 

135 

136 >>> invert_complex(exp(x), 1, x) 

137 (x, ImageSet(Lambda(_n, 2*_n*I*pi), Integers)) 

138 >>> invert_real(exp(x), 1, x) 

139 (x, {0}) 

140 

141 See Also 

142 ======== 

143 invert_real, invert_complex 

144 """ 

145 x = sympify(x) 

146 if not x.is_Symbol: 

147 raise ValueError("x must be a symbol") 

148 f_x = sympify(f_x) 

149 if x not in f_x.free_symbols: 

150 raise ValueError("Inverse of constant function doesn't exist") 

151 y = sympify(y) 

152 if x in y.free_symbols: 

153 raise ValueError("y should be independent of x ") 

154 

155 if domain.is_subset(S.Reals): 

156 x1, s = _invert_real(f_x, FiniteSet(y), x) 

157 else: 

158 x1, s = _invert_complex(f_x, FiniteSet(y), x) 

159 

160 if not isinstance(s, FiniteSet) or x1 != x: 

161 return x1, s 

162 

163 return x1, s.intersection(domain)

569 """Secondary helper to solve trigonometric equations, 

570 called when first helper fails """

577 numerators = [] 

578

583 except ValueError: 

584 raise ValueError("give up, we can't solve if this is not a polynomial in x")

608 y = Dummy('y') 

609 g, h = g.expand(), h.expand() 

610 g, h = g.subs(tan(x), y), h.subs(tan(x), y) 

611 

612 if g.has(x) or h.has(x): 

613 return ConditionSet(symbol, Eq(f_original, 0), domain) 

614 solns = solveset(g, y, S.Reals) - solveset(h, y, S.Reals) 

615 

616 if isinstance(solns, FiniteSet): 

617 result = Union(*[invert_real(tan(symbol/mu), s, symbol)[1] 

618 for s in solns]) 

619 dsol = invert_real(tan(symbol/mu), oo, symbol)[1] 

620 if degree(h) > degree(g): # If degree(denom)>degree(num) then there 

621 result = Union(result, dsol) # would be another sol at Lim(denom-->oo) 

622 return Intersection(result, domain) 

623 elif solns is S.EmptySet: 

624 return S.EmptySet 

625 else: 

626 return ConditionSet(symbol, Eq(f_original, 0), S.Reals)

629def _solve_as_poly(f, symbol, domain=S.Complexes): 

630 """ 

631 Solve the equation using polynomial techniques if it already is a 

632 polynomial equation or, with a change of variables, can be made so. 

633 """ 

634 result = None 

635 if f.is_polynomial(symbol): 

636 solns = roots(f, symbol, cubics=True, quartics=True, 

637 quintics=True, domain='EX') 

638 num_roots = sum(solns.values()) 

639 if degree(f, symbol) <= num_roots: 

640 result = FiniteSet(*solns.keys()) 

641 else: 

642 poly = Poly(f, symbol) 

643 solns = poly.all_roots() 

644 if poly.degree() <= len(solns): 

645 result = FiniteSet(*solns) 

646 else: 

647 result = ConditionSet(symbol, Eq(f, 0), domain) 

648 else: 

649 poly = Poly(f) 

650 if poly is None: 

651 result = ConditionSet(symbol, Eq(f, 0), domain) 

652 gens = [g for g in poly.gens if g.has(symbol)] 

653 

654 if len(gens) == 1: 

655 poly = Poly(poly, gens[0]) 

656 gen = poly.gen 

657 deg = poly.degree() 

658 poly = Poly(poly.as_expr(), poly.gen, composite=True) 

659 poly_solns = FiniteSet(*roots(poly, cubics=True, quartics=True, 

660 quintics=True).keys()) 

661 

662 if len(poly_solns) < deg: 

663 result = ConditionSet(symbol, Eq(f, 0), domain) 

664 

665 if gen != symbol: 

666 y = Dummy('y') 

667 inverter = invert_real if domain.is_subset(S.Reals) else invert_complex 

668 lhs, rhs_s = inverter(gen, y, symbol) 

669 if lhs == symbol: 

670 result = Union(*[rhs_s.subs(y, s) for s in poly_solns]) 

671 else: 

672 result = ConditionSet(symbol, Eq(f, 0), domain) 

673 else: 

674 result = ConditionSet(symbol, Eq(f, 0), domain) 

675 

676 if result is not None: 

677 if isinstance(result, FiniteSet): 

678 # this is to simplify solutions like -sqrt(-I) to sqrt(2)/2 

679 # - sqrt(2)*I/2. We are not expanding for solution with symbols 

680 # or undefined functions because that makes the solution more complicated. 

681 # For example, expand_complex(a) returns re(a) + I*im(a) 

682 if all([s.atoms(Symbol, AppliedUndef) == set() and not isinstance(s, RootOf) 

683 for s in result]): 

684 s = Dummy('s') 

685 result = imageset(Lambda(s, expand_complex(s)), result) 

686 if isinstance(result, FiniteSet): 

687 result = result.intersection(domain) 

688 return result 

689 else: 

690 return ConditionSet(symbol, Eq(f, 0), domain)

693def _has_rational_power(expr, symbol): 

694 """ 

695 Returns (bool, den) where bool is True if the term has a 

696 non-integer rational power and den is the denominator of the 

697 expression's exponent. 

698 

699 Examples 

700 ======== 

701 

702 >>> from sympy.solvers.solveset import _has_rational_power 

703 >>> from sympy import sqrt 

704 >>> from sympy.abc import x 

705 >>> _has_rational_power(sqrt(x), x) 

706 (True, 2) 

707 >>> _has_rational_power(x**2, x) 

708 (False, 1) 

709 """ 

710 a, p, q = Wild('a'), Wild('p'), Wild('q') 

711 pattern_match = expr.match(a*p**q) or {} 

712 if pattern_match.get(a, S.Zero) is S.Zero: 

713 return (False, S.One) 

714 elif p not in pattern_match.keys(): 

715 return (False, S.One) 

716 elif isinstance(pattern_match[q], Rational) \ 

717 and pattern_match[p].has(symbol): 

718 if not pattern_match[q].q == S.One: 

719 return (True, pattern_match[q].q) 

720 

721 if not isinstance(pattern_match[a], Pow) \ 

722 or isinstance(pattern_match[a], Mul): 

723 return (False, S.One) 

724 else: 

725 return _has_rational_power(pattern_match[a], symbol)

728def _solve_radical(f, symbol, solveset_solver): 

729 """ Helper function to solve equations with radicals """ 

730 eq, cov = unrad(f) 

731 if not cov: 

732 result = solveset_solver(eq, symbol) - \ 

733 Union(*[solveset_solver(g, symbol) for g in denoms(f, symbol)]) 

734 else: 

735 y, yeq = cov 

736 if not solveset_solver(y - I, y): 

737 yreal = Dummy('yreal', real=True) 

738 yeq = yeq.xreplace({y: yreal}) 

739 eq = eq.xreplace({y: yreal}) 

740 y = yreal 

741 g_y_s = solveset_solver(yeq, symbol) 

742 f_y_sols = solveset_solver(eq, y) 

743 result = Union(*[imageset(Lambda(y, g_y), f_y_sols) 

744 for g_y in g_y_s]) 

745 

746 if isinstance(result, Complement) or isinstance(result,ConditionSet): 

747 solution_set = result 

748 else: 

749 f_set = [] # solutions for FiniteSet 

750 c_set = [] # solutions for ConditionSet 

751 for s in result: 

752 if checksol(f, symbol, s): 

753 f_set.append(s) 

754 else: 

755 c_set.append(s) 

756 solution_set = FiniteSet(*f_set) + ConditionSet(symbol, Eq(f, 0), FiniteSet(*c_set)) 

757 

758 return solution_set

761def _solve_abs(f, symbol, domain): 

762 """ Helper function to solve equation involving absolute value function """ 

763 if not domain.is_subset(S.Reals): 

764 raise ValueError(filldedent(''' 

765 Absolute values cannot be inverted in the 

766 complex domain.''')) 

767 p, q, r = Wild('p'), Wild('q'), Wild('r') 

768 pattern_match = f.match(p*Abs(q) + r) or {} 

769 f_p, f_q, f_r = [pattern_match.get(i, S.Zero) for i in (p, q, r)] 

770

772 domain = continuous_domain(f_q, symbol, domain) 

773 q_pos_cond = solve_univariate_inequality(f_q >= 0, symbol,

782 else: 

783 return ConditionSet(symbol, Eq(f, 0), domain)

786def solve_decomposition(f, symbol, domain): 

787 """ 

788 Function to solve equations via the principle of "Decomposition 

789 and Rewriting". 

790 

791 Examples 

792 ======== 

793 >>> from sympy import exp, sin, Symbol, pprint, S 

794 >>> from sympy.solvers.solveset import solve_decomposition as sd 

795 >>> x = Symbol('x') 

796 >>> f1 = exp(2*x) - 3*exp(x) + 2 

797 >>> sd(f1, x, S.Reals) 

798 {0, log(2)} 

799 >>> f2 = sin(x)**2 + 2*sin(x) + 1 

800 >>> pprint(sd(f2, x, S.Reals), use_unicode=False) 

801 3*pi 

802 {2*n*pi + ---- | n in Integers} 

803 2 

804 >>> f3 = sin(x + 2) 

805 >>> pprint(sd(f3, x, S.Reals), use_unicode=False) 

806 {2*n*pi - 2 | n in Integers} U {2*n*pi - 2 + pi | n in Integers} 

807 

808 """ 

809 from sympy.solvers.decompogen import decompogen 

810 from sympy.calculus.util import function_range 

811 # decompose the given function 

812 g_s = decompogen(f, symbol) 

813 # `y_s` represents the set of values for which the function `g` is to be 

814 # solved. 

815 # `solutions` represent the solutions of the equations `g = y_s` or 

816 # `g = 0` depending on the type of `y_s`. 

817 # As we are interested in solving the equation: f = 0 

818 y_s = FiniteSet(0) 

819 for g in g_s: 

820 frange = function_range(g, symbol, domain) 

821 y_s = Intersection(frange, y_s) 

822 result = S.EmptySet 

823 if isinstance(y_s, FiniteSet): 

824 for y in y_s: 

825 solutions = solveset(Eq(g, y), symbol, domain) 

826 if not isinstance(solutions, ConditionSet): 

827 result += solutions 

828 

829 else: 

830 if isinstance(y_s, ImageSet): 

831 iter_iset = (y_s,) 

832 

833 elif isinstance(y_s, Union): 

834 iter_iset = y_s.args 

835 

836 elif isinstance(y_s, EmptySet): 

837 # y_s is not in the range of g in g_s, so no solution exists 

838 #in the given domain 

839 return y_s 

840 

841 for iset in iter_iset: 

842 new_solutions = solveset(Eq(iset.lamda.expr, g), symbol, domain) 

843 dummy_var = tuple(iset.lamda.expr.free_symbols)[0] 

844 base_set = iset.base_set 

845 if isinstance(new_solutions, FiniteSet): 

846 new_exprs = new_solutions 

847 

848 elif isinstance(new_solutions, Intersection): 

849 if isinstance(new_solutions.args[1], FiniteSet): 

850 new_exprs = new_solutions.args[1] 

851 

852 for new_expr in new_exprs: 

853 result += ImageSet(Lambda(dummy_var, new_expr), base_set) 

854 

855 if result is S.EmptySet: 

856 return ConditionSet(symbol, Eq(f, 0), domain) 

857 

858 y_s = result 

859 

860 return y_s

863def _solveset(f, symbol, domain, _check=False): 

864 """Helper for solveset to return a result from an expression 

865 that has already been sympify'ed and is known to contain the 

866 given symbol.""" 

867 # _check controls whether the answer is checked or not 

868 from sympy.simplify.simplify import signsimp 

869 

870 orig_f = f 

871 if f.is_Mul: 

872 coeff, f = f.as_independent(symbol, as_Add=False) 

873 if coeff in set([S.ComplexInfinity, S.NegativeInfinity, S.Infinity]): 

874 f = together(orig_f) 

875 elif f.is_Add: 

876 a, h = f.as_independent(symbol) 

877 m, h = h.as_independent(symbol, as_Add=False) 

878 if m not in set([S.ComplexInfinity, S.Zero, S.Infinity, 

879 S.NegativeInfinity]): 

880 f = a/m + h # XXX condition `m != 0` should be added to soln 

881 

882 # assign the solvers to use 

883 solver = lambda f, x, domain=domain: _solveset(f, x, domain) 

884 inverter = lambda f, rhs, symbol: _invert(f, rhs, symbol, domain) 

885 

886 result = EmptySet() 

887 

888 if f.expand().is_zero: 

889 return domain 

890 elif not f.has(symbol): 

891 return EmptySet() 

892 elif f.is_Mul and all(_is_finite_with_finite_vars(m, domain) 

893 for m in f.args): 

894 # if f(x) and g(x) are both finite we can say that the solution of 

895 # f(x)*g(x) == 0 is same as Union(f(x) == 0, g(x) == 0) is not true in 

896 # general. g(x) can grow to infinitely large for the values where 

897 # f(x) == 0. To be sure that we are not silently allowing any 

898 # wrong solutions we are using this technique only if both f and g are 

899 # finite for a finite input. 

900 result = Union(*[solver(m, symbol) for m in f.args]) 

901 elif _is_function_class_equation(TrigonometricFunction, f, symbol) or \ 

902 _is_function_class_equation(HyperbolicFunction, f, symbol): 

903 result = _solve_trig(f, symbol, domain) 

904 elif isinstance(f, arg): 

905 a = f.args[0] 

906 result = solveset_real(a > 0, symbol) 

907 elif f.is_Piecewise: 

908 result = EmptySet() 

909 expr_set_pairs = f.as_expr_set_pairs(domain) 

910 for (expr, in_set) in expr_set_pairs: 

911 if in_set.is_Relational: 

912 in_set = in_set.as_set() 

913 solns = solver(expr, symbol, in_set) 

914 result += solns 

915 elif isinstance(f, Eq): 

916 result = solver(Add(f.lhs, - f.rhs, evaluate=False), symbol, domain) 

917 

918 elif f.is_Relational: 

919 if not domain.is_subset(S.Reals): 

920 raise NotImplementedError(filldedent(''' 

921 Inequalities in the complex domain are 

922 not supported. Try the real domain by 

923 setting domain=S.Reals''')) 

924 try: 

925 result = solve_univariate_inequality( 

926 f, symbol, domain=domain, relational=False) 

927 except NotImplementedError: 

928 result = ConditionSet(symbol, f, domain) 

929 return result 

930 else: 

931 lhs, rhs_s = inverter(f, 0, symbol) 

932 if lhs == symbol: 

933 # do some very minimal simplification since 

934 # repeated inversion may have left the result 

935 # in a state that other solvers (e.g. poly) 

936 # would have simplified; this is done here 

937 # rather than in the inverter since here it 

938 # is only done once whereas there it would 

939 # be repeated for each step of the inversion 

940 if isinstance(rhs_s, FiniteSet): 

941 rhs_s = FiniteSet(*[Mul(* 

942 signsimp(i).as_content_primitive()) 

943 for i in rhs_s]) 

944 result = rhs_s 

945 

946 elif isinstance(rhs_s, FiniteSet): 

947 for equation in [lhs - rhs for rhs in rhs_s]: 

948 if equation == f: 

949 if any(_has_rational_power(g, symbol)[0] 

950 for g in equation.args) or _has_rational_power( 

951 equation, symbol)[0]: 

952 result += _solve_radical(equation, 

953 symbol, 

954 solver) 

955 elif equation.has(Abs): 

956 result += _solve_abs(f, symbol, domain) 

957 else: 

958 result_rational = _solve_as_rational(equation, symbol, domain) 

959 if isinstance(result_rational, ConditionSet): 

960 # may be a transcendental type equation 

961 result += _transolve(equation, symbol, domain) 

962 else: 

963 result += result_rational 

964 else: 

965 result += solver(equation, symbol) 

966 

967 elif rhs_s is not S.EmptySet: 

968 result = ConditionSet(symbol, Eq(f, 0), domain) 

969 

970 if isinstance(result, ConditionSet): 

971 num, den = f.as_numer_denom() 

972 if den.has(symbol): 

973 _result = _solveset(num, symbol, domain) 

974 if not isinstance(_result, ConditionSet): 

975 singularities = _solveset(den, symbol, domain) 

976 result = _result - singularities 

977 

978 if _check: 

979 if isinstance(result, ConditionSet): 

980 # it wasn't solved or has enumerated all conditions 

981 # -- leave it alone 

982 return result 

983 

984 # whittle away all but the symbol-containing core 

985 # to use this for testing 

986 fx = orig_f.as_independent(symbol, as_Add=True)[1] 

987 fx = fx.as_independent(symbol, as_Add=False)[1] 

988 

989 if isinstance(result, FiniteSet): 

990 # check the result for invalid solutions 

991 result = FiniteSet(*[s for s in result 

992 if isinstance(s, RootOf) 

993 or domain_check(fx, symbol, s)]) 

994 

995 return result

1026def _solve_exponential(lhs, rhs, symbol, domain): 

1027 r""" 

1028 Helper function for solving (supported) exponential equations. 

1029 

1030 Exponential equations are the sum of (currently) at most 

1031 two terms with one or both of them having a power with a 

1032 symbol-dependent exponent. 

1033 

1034 For example 

1035 

1036 .. math:: 5^{2x + 3} - 5^{3x - 1} 

1037 

1038 .. math:: 4^{5 - 9x} - e^{2 - x} 

1039 

1040 Parameters 

1041 ========== 

1042 

1043 lhs, rhs : Expr 

1044 The exponential equation to be solved, `lhs = rhs` 

1045 

1046 symbol : Symbol 

1047 The variable in which the equation is solved 

1048 

1049 domain : Set 

1050 A set over which the equation is solved. 

1051 

1052 Returns 

1053 ======= 

1054 

1055 A set of solutions satisfying the given equation. 

1056 A ``ConditionSet`` if the equation is unsolvable or 

1057 if the assumptions are not properly defined, in that case 

1058 a different style of ``ConditionSet`` is returned having the 

1059 solution(s) of the equation with the desired assumptions. 

1060 

1061 Examples 

1062 ======== 

1063 

1064 >>> from sympy.solvers.solveset import _solve_exponential as solve_expo 

1065 >>> from sympy import symbols, S 

1066 >>> x = symbols('x', real=True) 

1067 >>> a, b = symbols('a b') 

1068 >>> solve_expo(2**x + 3**x - 5**x, 0, x, S.Reals) # not solvable 

1069 ConditionSet(x, Eq(2**x + 3**x - 5**x, 0), Reals) 

1070 >>> solve_expo(a**x - b**x, 0, x, S.Reals) # solvable but incorrect assumptions 

1071 ConditionSet(x, (a > 0) & (b > 0), {0}) 

1072 >>> solve_expo(3**(2*x) - 2**(x + 3), 0, x, S.Reals) 

1073 {-3*log(2)/(-2*log(3) + log(2))} 

1074 >>> solve_expo(2**x - 4**x, 0, x, S.Reals) 

1075 {0} 

1076 

1077 * Proof of correctness of the method 

1078 

1079 The logarithm function is the inverse of the exponential function. 

1080 The defining relation between exponentiation and logarithm is: 

1081 

1082 .. math:: {\log_b x} = y \enspace if \enspace b^y = x 

1083 

1084 Therefore if we are given an equation with exponent terms, we can 

1085 convert every term to its corresponding logarithmic form. This is 

1086 achieved by taking logarithms and expanding the equation using 

1087 logarithmic identities so that it can easily be handled by ``solveset``. 

1088 

1089 For example: 

1090 

1091 .. math:: 3^{2x} = 2^{x + 3} 

1092 

1093 Taking log both sides will reduce the equation to 

1094 

1095 .. math:: (2x)\log(3) = (x + 3)\log(2) 

1096 

1097 This form can be easily handed by ``solveset``. 

1098 """ 

1099 unsolved_result = ConditionSet(symbol, Eq(lhs - rhs), domain) 

1100 newlhs = powdenest(lhs) 

1101 if lhs != newlhs: 

1102 # it may also be advantageous to factor the new expr 

1103 return _solveset(factor(newlhs - rhs), symbol, domain) # try again with _solveset 

1104 

1105 if not (isinstance(lhs, Add) and len(lhs.args) == 2): 

1106 # solving for the sum of more than two powers is possible 

1107 # but not yet implemented 

1108 return unsolved_result 

1109 

1110 if rhs != 0: 

1111 return unsolved_result 

1112 

1113 a, b = list(ordered(lhs.args)) 

1114 a_term = a.as_independent(symbol)[1] 

1115 b_term = b.as_independent(symbol)[1] 

1116 

1117 a_base, a_exp = a_term.base, a_term.exp 

1118 b_base, b_exp = b_term.base, b_term.exp 

1119 

1120 from sympy.functions.elementary.complexes import im 

1121 

1122 if domain.is_subset(S.Reals): 

1123 conditions = And( 

1124 a_base > 0, 

1125 b_base > 0, 

1126 Eq(im(a_exp), 0), 

1127 Eq(im(b_exp), 0)) 

1128 else: 

1129 conditions = And( 

1130 Ne(a_base, 0), 

1131 Ne(b_base, 0)) 

1132 

1133 L, R = map(lambda i: expand_log(log(i), force=True), (a, -b)) 

1134 solutions = _solveset(L - R, symbol, domain) 

1135 

1136 return ConditionSet(symbol, conditions, solutions)

1221 A ``ConditionSet`` if the equation is unsolvable. 

1222

1225 

1226 >>> from sympy import symbols, log, S 

1227 >>> from sympy.solvers.solveset import _solve_logarithm as solve_log 

1228 >>> x = symbols('x') 

1229 >>> f = log(x - 3) + log(x + 3) 

1230 >>> solve_log(f, 0, x, S.Reals) 

1231 {-sqrt(10), sqrt(10)} 

1232 

1233 * Proof of correctness 

1234 

1235 A logarithm is another way to write exponent and is defined by 

1236 

1237 .. math:: {\log_b x} = y \enspace if \enspace b^y = x 

1238 

1239 When one side of the equation contains a single logarithm, the 

1240 equation can be solved by rewriting the equation as an equivalent 

1241 exponential equation as defined above. But if one side contains 

1242 more than one logarithm, we need to use the properties of logarithm 

1243 to condense it into a single logarithm. 

1244 

1245 Take for example 

1246 

1247 .. math:: \log(2x) - 15 = 0 

1248 

1249 contains single logarithm, therefore we can directly rewrite it to 

1250 exponential form as 

1251 

1252 .. math:: x = \frac{e^{15}}{2} 

1253 

1254 But if the equation has more than one logarithm as 

1255 

1256 .. math:: \log(x - 3) + \log(x + 3) = 0 

1257 

1258 we use logarithmic identities to convert it into a reduced form 

1259 

1260 Using, 

1261 

1262 .. math:: \log(a) + \log(b) = \log(ab) 

1263 

1264 the equation becomes, 

1265 

1266 .. math:: \log((x - 3)(x + 3)) 

1267 

1268 This equation contains one logarithm and can be solved by rewriting 

1269 to exponents. 

1270 """ 

1271 new_lhs = logcombine(lhs, force=True) 

1272 new_f = new_lhs - rhs 

1273 

1274 return _solveset(new_f, symbol, domain)

1335def _transolve(f, symbol, domain): 

1336 r""" 

1337 Function to solve transcendental equations. It is a helper to 

1338 ``solveset`` and should be used internally. ``_transolve`` 

1339 currently supports the following class of equations: 

1340 

1341 - Exponential equations 

1342 - Logarithmic equations 

1343 

1344 Parameters 

1345 ========== 

1346 

1347 f : Any transcendental equation that needs to be solved. 

1348 This needs to be an expression, which is assumed 

1349 to be equal to ``0``. 

1350 

1351 symbol : The variable for which the equation is solved. 

1352 This needs to be of class ``Symbol``. 

1353 

1354 domain : A set over which the equation is solved. 

1355 This needs to be of class ``Set``. 

1356 

1357 Returns 

1358 ======= 

1359 

1360 Set 

1361 A set of values for ``symbol`` for which ``f`` is equal to 

1362 zero. An ``EmptySet`` is returned if ``f`` does not have solutions 

1363 in respective domain. A ``ConditionSet`` is returned as unsolved 

1364 object if algorithms to evaluate complete solution are not 

1365 yet implemented. 

1366 

1367 How to use ``_transolve`` 

1368 ========================= 

1369 

1370 ``_transolve`` should not be used as an independent function, because 

1371 it assumes that the equation (``f``) and the ``symbol`` comes from 

1372 ``solveset`` and might have undergone a few modification(s). 

1373 To use ``_transolve`` as an independent function the equation (``f``) 

1374 and the ``symbol`` should be passed as they would have been by 

1375 ``solveset``. 

1376 

1377 Examples 

1378 ======== 

1379 

1380 >>> from sympy.solvers.solveset import _transolve as transolve 

1381 >>> from sympy.solvers.solvers import _tsolve as tsolve 

1382 >>> from sympy import symbols, S, pprint 

1383 >>> x = symbols('x', real=True) # assumption added 

1384 >>> transolve(5**(x - 3) - 3**(2*x + 1), x, S.Reals) 

1385 {-(log(3) + 3*log(5))/(-log(5) + 2*log(3))} 

1386 

1387 How ``_transolve`` works 

1388 ======================== 

1389 

1390 ``_transolve`` uses two types of helper functions to solve equations 

1391 of a particular class: 

1392 

1393 Identifying helpers: To determine whether a given equation 

1394 belongs to a certain class of equation or not. Returns either 

1395 ``True`` or ``False``. 

1396 

1397 Solving helpers: Once an equation is identified, a corresponding 

1398 helper either solves the equation or returns a form of the equation 

1399 that ``solveset`` might better be able to handle. 

1400 

1401 * Philosophy behind the module 

1402 

1403 The purpose of ``_transolve`` is to take equations which are not 

1404 already polynomial in their generator(s) and to either recast them 

1405 as such through a valid transformation or to solve them outright. 

1406 A pair of helper functions for each class of supported 

1407 transcendental functions are employed for this purpose. One 

1408 identifies the transcendental form of an equation and the other 

1409 either solves it or recasts it into a tractable form that can be 

1410 solved by ``solveset``. 

1411 For example, an equation in the form `ab^{f(x)} - cd^{g(x)} = 0` 

1412 can be transformed to 

1413 `\log(a) + f(x)\log(b) - \log(c) - g(x)\log(d) = 0` 

1414 (under certain assumptions) and this can be solved with ``solveset`` 

1415 if `f(x)` and `g(x)` are in polynomial form. 

1416 

1417 How ``_transolve`` is better than ``_tsolve`` 

1418 ============================================= 

1419 

1420 1) Better output 

1421 

1422 ``_transolve`` provides expressions in a more simplified form. 

1423 

1424 Consider a simple exponential equation 

1425 

1426 >>> f = 3**(2*x) - 2**(x + 3) 

1427 >>> pprint(transolve(f, x, S.Reals), use_unicode=False) 

1428 -3*log(2) 

1429 {------------------} 

1430 -2*log(3) + log(2) 

1431 >>> pprint(tsolve(f, x), use_unicode=False) 

1432 / 3 \ 

1433 | --------| 

1434 | log(2/9)| 

1435 [-log\2 /] 

1436 

1437 2) Extensible 

1438 

1439 The API of ``_transolve`` is designed such that it is easily 

1440 extensible, i.e. the code that solves a given class of 

1441 equations is encapsulated in a helper and not mixed in with 

1442 the code of ``_transolve`` itself. 

1443 

1444 3) Modular 

1445 

1446 ``_transolve`` is designed to be modular i.e, for every class of 

1447 equation a separate helper for identification and solving is 

1448 implemented. This makes it easy to change or modify any of the 

1449 method implemented directly in the helpers without interfering 

1450 with the actual structure of the API. 

1451 

1452 4) Faster Computation 

1453 

1454 Solving equation via ``_transolve`` is much faster as compared to 

1455 ``_tsolve``. In ``solve``, attempts are made computing every possibility 

1456 to get the solutions. This series of attempts makes solving a bit 

1457 slow. In ``_transolve``, computation begins only after a particular 

1458 type of equation is identified. 

1459 

1460 How to add new class of equations 

1461 ================================= 

1462 

1463 Adding a new class of equation solver is a three-step procedure: 

1464 

1465 - Identify the type of the equations 

1466 

1467 Determine the type of the class of equations to which they belong: 

1468 it could be of ``Add``, ``Pow``, etc. types. Separate internal functions 

1469 are used for each type. Write identification and solving helpers 

1470 and use them from within the routine for the given type of equation 

1471 (after adding it, if necessary). Something like: 

1472 

1473 .. code-block:: python 

1474 

1475 def add_type(lhs, rhs, x): 

1476 .... 

1477 if _is_exponential(lhs, x): 

1478 new_eq = _solve_exponential(lhs, rhs, x) 

1479 .... 

1480 rhs, lhs = eq.as_independent(x) 

1481 if lhs.is_Add: 

1482 result = add_type(lhs, rhs, x) 

1483 

1484 - Define the identification helper. 

1485 

1486 - Define the solving helper. 

1487 

1488 Apart from this, a few other things needs to be taken care while 

1489 adding an equation solver: 

1490 

1491 - Naming conventions: 

1492 Name of the identification helper should be as 

1493 ``_is_class`` where class will be the name or abbreviation 

1494 of the class of equation. The solving helper will be named as 

1495 ``_solve_class``. 

1496 For example: for exponential equations it becomes 

1497 ``_is_exponential`` and ``_solve_expo``. 

1498 - The identifying helpers should take two input parameters, 

1499 the equation to be checked and the variable for which a solution 

1500 is being sought, while solving helpers would require an additional 

1501 domain parameter. 

1502 - Be sure to consider corner cases. 

1503 - Add tests for each helper. 

1504 - Add a docstring to your helper that describes the method 

1505 implemented. 

1506 The documentation of the helpers should identify: 

1507 

1508 - the purpose of the helper, 

1509 - the method used to identify and solve the equation, 

1510 - a proof of correctness 

1511 - the return values of the helpers 

1512 """

1552 

1553 Parameters 

1554 ========== 

1555 

1556 f : Expr or a relational. 

1557 The target equation or inequality 

1558 symbol : Symbol

1569 A `ConditionSet` is returned as unsolved object if algorithms 

1570 to evaluate complete solution are not yet implemented.

1573

1575 ====== 

1576

1578 The algorithms to solve inequalities in complex domain are 

1579 not yet implemented. 

1580 ValueError

1582 RuntimeError

1584 

1585

1587 ===== 

1588

1590 in this function they refer to solutions of an expression. So 0 and 1 

1591 return the Domain and EmptySet, respectively, while True and False 

1592 return the opposite (as they are assumed to be solutions of relational 

1593 expressions). 

1594 

1595

1598 

1599 solveset_real: solver for real domain 

1600 solveset_complex: solver for complex domain

1606 >>> from sympy.solvers.solveset import solveset, solveset_real

1609 to the solving of the equation in the complex domain (and this 

1610 is not affected by the assumptions on the symbol): 

1611

1613 >>> pprint(solveset(exp(x) - 1, x), use_unicode=False)

1623

1625 >>> x = Symbol('x') 

1626 >>> solveset(exp(x) - 1, x, R) 

1627 {0} 

1628 >>> solveset_real(exp(x) - 1, x) 

1629 {0} 

1630

1632 but there may be some slight difference: 

1633 

1634 >>> pprint(solveset(sin(x)/x,x), use_unicode=False)

1664 raise ValueError("%s is not a valid domain" %(domain))

1725def solvify(f, symbol, domain): 

1726 """Solves an equation using solveset and returns the solution in accordance 

1727 with the `solve` output API. 

1728 

1729 Returns 

1730 ======= 

1731 

1732 We classify the output based on the type of solution returned by `solveset`. 

1733 

1734 Solution | Output 

1735 ---------------------------------------- 

1736 FiniteSet | list 

1737 

1738 ImageSet, | list (if `f` is periodic) 

1739 Union | 

1740 

1741 EmptySet | empty list 

1742 

1743 Others | None 

1744 

1745 

1746 Raises 

1747 ====== 

1748 

1749 NotImplementedError 

1750 A ConditionSet is the input. 

1751 

1752 Examples 

1753 ======== 

1754 

1755 >>> from sympy.solvers.solveset import solvify, solveset 

1756 >>> from sympy.abc import x 

1757 >>> from sympy import S, tan, sin, exp 

1758 >>> solvify(x**2 - 9, x, S.Reals) 

1759 [-3, 3] 

1760 >>> solvify(sin(x) - 1, x, S.Reals) 

1761 [pi/2] 

1762 >>> solvify(tan(x), x, S.Reals) 

1763 [0] 

1764 >>> solvify(exp(x) - 1, x, S.Complexes) 

1765 

1766 >>> solvify(exp(x) - 1, x, S.Reals) 

1767 [0] 

1768 

1769 """ 

1770 solution_set = solveset(f, symbol, domain) 

1771 result = None 

1772 if solution_set is S.EmptySet: 

1773 result = [] 

1774 

1775 elif isinstance(solution_set, ConditionSet): 

1776 raise NotImplementedError('solveset is unable to solve this equation.') 

1777 

1778 elif isinstance(solution_set, FiniteSet): 

1779 result = list(solution_set) 

1780 

1781 else: 

1782 period = periodicity(f, symbol) 

1783 if period is not None: 

1784 solutions = S.EmptySet 

1785 iter_solutions = () 

1786 if isinstance(solution_set, ImageSet): 

1787 iter_solutions = (solution_set,) 

1788 elif isinstance(solution_set, Union): 

1789 if all(isinstance(i, ImageSet) for i in solution_set.args): 

1790 iter_solutions = solution_set.args 

1791 

1792 for solution in iter_solutions: 

1793 solutions += solution.intersect(Interval(0, period, False, True)) 

1794 

1795 if isinstance(solutions, FiniteSet): 

1796 result = list(solutions) 

1797 

1798 else: 

1799 solution = solution_set.intersect(domain) 

1800 if isinstance(solution, FiniteSet): 

1801 result += solution 

1802 

1803 return result

1811def linear_coeffs(eq, *syms, **_kw): 

1812 """Return a list whose elements are the coefficients of the 

1813 corresponding symbols in the sum of terms in ``eq``. 

1814 The additive constant is returned as the last element of the 

1815 list. 

1816 

1817 Examples 

1818 ======== 

1819 

1820 >>> from sympy.solvers.solveset import linear_coeffs 

1821 >>> from sympy.abc import x, y, z 

1822 

1823 >>> linear_coeffs(3*x + 2*y - 1, x, y) 

1824 [3, 2, -1] 

1825 

1826 It is not necessary to expand the expression: 

1827 

1828 >>> linear_coeffs(x + y*(z*(x*3 + 2) + 3), x) 

1829 [3*y*z + 1, y*(2*z + 3)] 

1830 

1831 But if there are nonlinear or cross terms -- even if they would 

1832 cancel after simplification -- an error is raised so the situation 

1833 does not pass silently past the caller's attention: 

1834 

1835 >>> eq = 1/x*(x - 1) + 1/x 

1836 >>> linear_coeffs(eq.expand(), x) 

1837 [0, 1] 

1838 >>> linear_coeffs(eq, x) 

1839 Traceback (most recent call last): 

1840 ... 

1841 ValueError: nonlinear term encountered: 1/x 

1842 

1843 >>> linear_coeffs(x*(y + 1) - x*y, x, y) 

1844 Traceback (most recent call last): 

1845 ... 

1846 ValueError: nonlinear term encountered: x*(y + 1) 

1847 """ 

1848 d = defaultdict(list) 

1849 c, terms = _sympify(eq).as_coeff_add(*syms) 

1850 d[0].extend(Add.make_args(c)) 

1851 for t in terms: 

1852 m, f = t.as_coeff_mul(*syms) 

1853 if len(f) != 1: 

1854 break 

1855 f = f[0] 

1856 if f in syms: 

1857 d[f].append(m) 

1858 elif f.is_Add: 

1859 d1 = linear_coeffs(f, *syms, **{'dict': True}) 

1860 d[0].append(m*d1.pop(0)) 

1861 for xf, vf in d1.items(): 

1862 d[xf].append(m*vf) 

1863 else: 

1864 break 

1865 else: 

1866 for k, v in d.items(): 

1867 d[k] = Add(*v) 

1868 if not _kw: 

1869 return [d.get(s, S.Zero) for s in syms] + [d[0]] 

1870 return d # default is still list but this won't matter 

1871 raise ValueError('nonlinear term encountered: %s' % t)

2851def nonlinsolve(system, *symbols): 

2852 r""" 

2853 Solve system of N non linear equations with M variables, which means both 

2854 under and overdetermined systems are supported. Positive dimensional 

2855 system is also supported (A system with infinitely many solutions is said 

2856 to be positive-dimensional). In Positive dimensional system solution will 

2857 be dependent on at least one symbol. Returns both real solution 

2858 and complex solution(If system have). The possible number of solutions 

2859 is zero, one or infinite. 

2860 

2861 Parameters 

2862 ========== 

2863 

2864 system : list of equations 

2865 The target system of equations 

2866 symbols : list of Symbols 

2867 symbols should be given as a sequence eg. list 

2868 

2869 Returns 

2870 ======= 

2871 

2872 A FiniteSet of ordered tuple of values of `symbols` for which the `system` 

2873 has solution. Order of values in the tuple is same as symbols present in 

2874 the parameter `symbols`. 

2875 

2876 Please note that general FiniteSet is unordered, the solution returned 

2877 here is not simply a FiniteSet of solutions, rather it is a FiniteSet of 

2878 ordered tuple, i.e. the first & only argument to FiniteSet is a tuple of 

2879 solutions, which is ordered, & hence the returned solution is ordered. 

2880 

2881 Also note that solution could also have been returned as an ordered tuple, 

2882 FiniteSet is just a wrapper `{}` around the tuple. It has no other 

2883 significance except for the fact it is just used to maintain a consistent 

2884 output format throughout the solveset. 

2885 

2886 For the given set of Equations, the respective input types 

2887 are given below: 

2888 

2889 .. math:: x*y - 1 = 0 

2890 .. math:: 4*x**2 + y**2 - 5 = 0 

2891 

2892 `system = [x*y - 1, 4*x**2 + y**2 - 5]` 

2893 `symbols = [x, y]` 

2894 

2895 Raises 

2896 ====== 

2897 

2898 ValueError 

2899 The input is not valid. 

2900 The symbols are not given. 

2901 AttributeError 

2902 The input symbols are not `Symbol` type. 

2903 

2904 Examples 

2905 ======== 

2906 

2907 >>> from sympy.core.symbol import symbols 

2908 >>> from sympy.solvers.solveset import nonlinsolve 

2909 >>> x, y, z = symbols('x, y, z', real=True) 

2910 >>> nonlinsolve([x*y - 1, 4*x**2 + y**2 - 5], [x, y]) 

2911 {(-1, -1), (-1/2, -2), (1/2, 2), (1, 1)} 

2912 

2913 1. Positive dimensional system and complements: 

2914 

2915 >>> from sympy import pprint 

2916 >>> from sympy.polys.polytools import is_zero_dimensional 

2917 >>> a, b, c, d = symbols('a, b, c, d', real=True) 

2918 >>> eq1 = a + b + c + d 

2919 >>> eq2 = a*b + b*c + c*d + d*a 

2920 >>> eq3 = a*b*c + b*c*d + c*d*a + d*a*b 

2921 >>> eq4 = a*b*c*d - 1 

2922 >>> system = [eq1, eq2, eq3, eq4] 

2923 >>> is_zero_dimensional(system) 

2924 False 

2925 >>> pprint(nonlinsolve(system, [a, b, c, d]), use_unicode=False) 

2926 -1 1 1 -1 

2927 {(---, -d, -, {d} \ {0}), (-, -d, ---, {d} \ {0})} 

2928 d d d d 

2929 >>> nonlinsolve([(x+y)**2 - 4, x + y - 2], [x, y]) 

2930 {(2 - y, y)} 

2931 

2932 2. If some of the equations are non polynomial equation then `nonlinsolve` 

2933 will call `substitution` function and returns real and complex solutions, 

2934 if present. 

2935 

2936 >>> from sympy import exp, sin 

2937 >>> nonlinsolve([exp(x) - sin(y), y**2 - 4], [x, y]) 

2938 {(log(sin(2)), 2), (ImageSet(Lambda(_n, I*(2*_n*pi + pi) + 

2939 log(sin(2))), Integers), -2), (ImageSet(Lambda(_n, 2*_n*I*pi + 

2940 Mod(log(sin(2)), 2*I*pi)), Integers), 2)} 

2941 

2942 3. If system is Non linear polynomial zero dimensional then it returns 

2943 both solution (real and complex solutions, if present using 

2944 `solve_poly_system`): 

2945 

2946 >>> from sympy import sqrt 

2947 >>> nonlinsolve([x**2 - 2*y**2 -2, x*y - 2], [x, y]) 

2948 {(-2, -1), (2, 1), (-sqrt(2)*I, sqrt(2)*I), (sqrt(2)*I, -sqrt(2)*I)} 

2949 

2950 4. `nonlinsolve` can solve some linear(zero or positive dimensional) 

2951 system (because it is using `groebner` function to get the 

2952 groebner basis and then `substitution` function basis as the new `system`). 

2953 But it is not recommended to solve linear system using `nonlinsolve`, 

2954 because `linsolve` is better for all kind of linear system. 

2955 

2956 >>> nonlinsolve([x + 2*y -z - 3, x - y - 4*z + 9 , y + z - 4], [x, y, z]) 

2957 {(3*z - 5, 4 - z, z)} 

2958 

2959 5. System having polynomial equations and only real solution is present 

2960 (will be solved using `solve_poly_system`): 

2961 

2962 >>> e1 = sqrt(x**2 + y**2) - 10 

2963 >>> e2 = sqrt(y**2 + (-x + 10)**2) - 3 

2964 >>> nonlinsolve((e1, e2), (x, y)) 

2965 {(191/20, -3*sqrt(391)/20), (191/20, 3*sqrt(391)/20)} 

2966 >>> nonlinsolve([x**2 + 2/y - 2, x + y - 3], [x, y]) 

2967 {(1, 2), (1 - sqrt(5), 2 + sqrt(5)), (1 + sqrt(5), 2 - sqrt(5))} 

2968 >>> nonlinsolve([x**2 + 2/y - 2, x + y - 3], [y, x]) 

2969 {(2, 1), (2 - sqrt(5), 1 + sqrt(5)), (2 + sqrt(5), 1 - sqrt(5))} 

2970 

2971 6. It is better to use symbols instead of Trigonometric Function or 

2972 Function (e.g. replace `sin(x)` with symbol, replace `f(x)` with symbol 

2973 and so on. Get soln from `nonlinsolve` and then using `solveset` get 

2974 the value of `x`) 

2975 

2976 How nonlinsolve is better than old solver `_solve_system` : 

2977 =========================================================== 

2978 

2979 1. A positive dimensional system solver : nonlinsolve can return 

2980 solution for positive dimensional system. It finds the 

2981 Groebner Basis of the positive dimensional system(calling it as 

2982 basis) then we can start solving equation(having least number of 

2983 variable first in the basis) using solveset and substituting that 

2984 solved solutions into other equation(of basis) to get solution in 

2985 terms of minimum variables. Here the important thing is how we 

2986 are substituting the known values and in which equations. 

2987 

2988 2. Real and Complex both solutions : nonlinsolve returns both real 

2989 and complex solution. If all the equations in the system are polynomial 

2990 then using `solve_poly_system` both real and complex solution is returned. 

2991 If all the equations in the system are not polynomial equation then goes to 

2992 `substitution` method with this polynomial and non polynomial equation(s), 

2993 to solve for unsolved variables. Here to solve for particular variable 

2994 solveset_real and solveset_complex is used. For both real and complex 

2995 solution function `_solve_using_know_values` is used inside `substitution` 

2996 function.(`substitution` function will be called when there is any non 

2997 polynomial equation(s) is present). When solution is valid then add its 

2998 general solution in the final result. 

2999 

3000 3. Complement and Intersection will be added if any : nonlinsolve maintains 

3001 dict for complements and Intersections. If solveset find complements or/and 

3002 Intersection with any Interval or set during the execution of 

3003 `substitution` function ,then complement or/and Intersection for that 

3004 variable is added before returning final solution. 

3005 

3006 """

3008 from sympy.polys import RR

3017

3019 symbols = symbols[0]

3022 msg = ('Symbols must be given, for which solution of the ' 

3023 'system is to be found.') 

3024 raise IndexError(filldedent(msg)) 

3025 

3026 system, symbols, swap = recast_to_symbols(system, symbols) 

3027 if swap: 

3028 soln = nonlinsolve(system, symbols) 

3029 return FiniteSet(*[tuple(i.xreplace(swap) for i in s) for s in soln]) 

3030 

3031 if len(system) == 1 and len(symbols) == 1: 

3032 return _solveset_work(system, symbols) 

3033

3035 polys, polys_expr, nonpolys, denominators = _separate_poly_nonpoly(
