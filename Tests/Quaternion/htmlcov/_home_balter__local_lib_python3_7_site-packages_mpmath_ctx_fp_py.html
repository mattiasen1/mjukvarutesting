199 def hypsum(ctx, p, q, types, coeffs, z, maxterms=6000, **kwargs): 

200 coeffs = list(coeffs) 

201 num = range(p) 

202 den = range(p,p+q) 

203 tol = ctx.eps 

204 s = t = 1.0 

205 k = 0 

206 while 1: 

207 for i in num: t *= (coeffs[i]+k) 

208 for i in den: t /= (coeffs[i]+k) 

209 k += 1; t /= k; t *= z; s += t
