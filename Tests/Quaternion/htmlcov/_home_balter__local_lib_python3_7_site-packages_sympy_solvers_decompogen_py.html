7def decompogen(f, symbol): 

8 """ 

9 Computes General functional decomposition of ``f``. 

10 Given an expression ``f``, returns a list ``[f_1, f_2, ..., f_n]``, 

11 where:: 

12 f = f_1 o f_2 o ... f_n = f_1(f_2(... f_n)) 

13 

14 Note: This is a General decomposition function. It also decomposes 

15 Polynomials. For only Polynomial decomposition see ``decompose`` in polys. 

16 

17 Examples 

18 ======== 

19 

20 >>> from sympy.solvers.decompogen import decompogen 

21 >>> from sympy.abc import x 

22 >>> from sympy import sqrt, sin, cos 

23 >>> decompogen(sin(cos(x)), x) 

24 [sin(x), cos(x)] 

25 >>> decompogen(sin(x)**2 + sin(x) + 1, x) 

26 [x**2 + x + 1, sin(x)] 

27 >>> decompogen(sqrt(6*x**2 - 5), x) 

28 [sqrt(x), 6*x**2 - 5] 

29 >>> decompogen(sin(sqrt(cos(x**2 + 1))), x) 

30 [sin(x), sqrt(x), cos(x), x**2 + 1] 

31 >>> decompogen(x**4 + 2*x**3 - x - 1, x) 

32 [x**2 - x - 1, x**2 + x] 

33 

34 """ 

35 f = sympify(f) 

36 if not isinstance(f, Expr) or isinstance(f, Relational): 

37 raise TypeError('expecting Expr but got: `%s`' % func_name(f)) 

38 if symbol not in f.free_symbols: 

39 return [f] 

40 

41 result = [] 

42 

43 # ===== Simple Functions ===== # 

44 if isinstance(f, (Function, Pow)): 

45 if f.args[0] == symbol: 

46 return [f] 

47 result += [f.subs(f.args[0], symbol)] + decompogen(f.args[0], symbol) 

48 return result 

49 

50 # ===== Convert to Polynomial ===== # 

51 fp = Poly(f) 

52 gens = list(filter(lambda x: symbol in x.free_symbols , fp.gens)) 

53 

54 if len(gens) == 1 and gens[0] != symbol: 

55 f1 = f.subs(gens[0], symbol) 

56 f2 = gens[0] 

57 result += [f1] + decompogen(f2, symbol) 

58 return result 

59 

60 # ===== Polynomial decompose() ====== # 

61 try: 

62 result += decompose(f) 

63 return result 

64 except ValueError: 

65 return [f]

89 sqrt(6*x**2 - 5)

91 sin(sqrt(cos(x**2 + 1))) 

92 >>> compogen([x**2 - x - 1, x**2 + x], x) 

93 -x**2 - x + (x**2 + x)**2 - 1 

94 """

96 return g_s[0]

100 if len(g_s) == 2:
