10def polyval(ctx, coeffs, x, derivative=False): 

11 r""" 

12 Given coefficients `[c_n, \ldots, c_2, c_1, c_0]` and a number `x`, 

13 :func:`~mpmath.polyval` evaluates the polynomial 

14 

15 .. math :: 

16 

17 P(x) = c_n x^n + \ldots + c_2 x^2 + c_1 x + c_0. 

18 

19 If *derivative=True* is set, :func:`~mpmath.polyval` simultaneously 

20 evaluates `P(x)` with the derivative, `P'(x)`, and returns the 

21 tuple `(P(x), P'(x))`. 

22 

23 >>> from mpmath import * 

24 >>> mp.pretty = True 

25 >>> polyval([3, 0, 2], 0.5) 

26 2.75 

27 >>> polyval([3, 0, 2], 0.5, derivative=True) 

28 (2.75, 3.0) 

29 

30 The coefficients and the evaluation point may be any combination 

31 of real or complex numbers. 

32 """ 

33 if not coeffs: 

34 return ctx.zero 

35 p = ctx.convert(coeffs[0]) 

36 q = ctx.zero 

37 for c in coeffs[1:]: 

38 if derivative: 

39 q = p + x*q 

40 p = c + x*p 

41 if derivative: 

42 return p, q 

43 else: 

44 return p

46@defun 

47def polyroots(ctx, coeffs, maxsteps=50, cleanup=True, extraprec=10, 

48 error=False, roots_init=None): 

49 """ 

50 Computes all roots (real or complex) of a given polynomial. 

51 

52 The roots are returned as a sorted list, where real roots appear first 

53 followed by complex conjugate roots as adjacent elements. The polynomial 

54 should be given as a list of coefficients, in the format used by 

55 :func:`~mpmath.polyval`. The leading coefficient must be nonzero. 

56 

57 With *error=True*, :func:`~mpmath.polyroots` returns a tuple *(roots, err)* 

58 where *err* is an estimate of the maximum error among the computed roots. 

59 

60 **Examples** 

61 

62 Finding the three real roots of `x^3 - x^2 - 14x + 24`:: 

63 

64 >>> from mpmath import * 

65 >>> mp.dps = 15; mp.pretty = True 

66 >>> nprint(polyroots([1,-1,-14,24]), 4) 

67 [-4.0, 2.0, 3.0] 

68 

69 Finding the two complex conjugate roots of `4x^2 + 3x + 2`, with an 

70 error estimate:: 

71 

72 >>> roots, err = polyroots([4,3,2], error=True) 

73 >>> for r in roots: 

74 ... print(r) 

75 ... 

76 (-0.375 + 0.59947894041409j) 

77 (-0.375 - 0.59947894041409j) 

78 >>> 

79 >>> err 

80 2.22044604925031e-16 

81 >>> 

82 >>> polyval([4,3,2], roots[0]) 

83 (2.22044604925031e-16 + 0.0j) 

84 >>> polyval([4,3,2], roots[1]) 

85 (2.22044604925031e-16 + 0.0j) 

86 

87 The following example computes all the 5th roots of unity; that is, 

88 the roots of `x^5 - 1`:: 

89 

90 >>> mp.dps = 20 

91 >>> for r in polyroots([1, 0, 0, 0, 0, -1]): 

92 ... print(r) 

93 ... 

94 1.0 

95 (-0.8090169943749474241 + 0.58778525229247312917j) 

96 (-0.8090169943749474241 - 0.58778525229247312917j) 

97 (0.3090169943749474241 + 0.95105651629515357212j) 

98 (0.3090169943749474241 - 0.95105651629515357212j) 

99 

100 **Precision and conditioning** 

101 

102 The roots are computed to the current working precision accuracy. If this 

103 accuracy cannot be achieved in `maxsteps` steps, then a `NoConvergence` 

104 exception is raised. The algorithm internally is using the current working 

105 precision extended by `extraprec`. If `NoConvergence` was raised, that is 

106 caused either by not having enough extra precision to achieve convergence 

107 (in which case increasing `extraprec` should fix the problem) or too low 

108 `maxsteps` (in which case increasing `maxsteps` should fix the problem), or 

109 a combination of both. 

110 

111 The user should always do a convergence study with regards to `extraprec` 

112 to ensure accurate results. It is possible to get convergence to a wrong 

113 answer with too low `extraprec`. 

114 

115 Provided there are no repeated roots, :func:`~mpmath.polyroots` can 

116 typically compute all roots of an arbitrary polynomial to high precision:: 

117 

118 >>> mp.dps = 60 

119 >>> for r in polyroots([1, 0, -10, 0, 1]): 

120 ... print r 

121 ... 

122 -3.14626436994197234232913506571557044551247712918732870123249 

123 -0.317837245195782244725757617296174288373133378433432554879127 

124 0.317837245195782244725757617296174288373133378433432554879127 

125 3.14626436994197234232913506571557044551247712918732870123249 

126 >>> 

127 >>> sqrt(3) + sqrt(2) 

128 3.14626436994197234232913506571557044551247712918732870123249 

129 >>> sqrt(3) - sqrt(2) 

130 0.317837245195782244725757617296174288373133378433432554879127 

131 

132 **Algorithm** 

133 

134 :func:`~mpmath.polyroots` implements the Durand-Kerner method [1], which 

135 uses complex arithmetic to locate all roots simultaneously. 

136 The Durand-Kerner method can be viewed as approximately performing 

137 simultaneous Newton iteration for all the roots. In particular, 

138 the convergence to simple roots is quadratic, just like Newton's 

139 method. 

140 

141 Although all roots are internally calculated using complex arithmetic, any 

142 root found to have an imaginary part smaller than the estimated numerical 

143 error is truncated to a real number (small real parts are also chopped). 

144 Real roots are placed first in the returned list, sorted by value. The 

145 remaining complex roots are sorted by their real parts so that conjugate 

146 roots end up next to each other. 

147 

148 **References** 

149 

150 1. http://en.wikipedia.org/wiki/Durand-Kerner_method 

151 

152 """ 

153 if len(coeffs) <= 1: 

154 if not coeffs or not coeffs[0]: 

155 raise ValueError("Input to polyroots must not be the zero polynomial") 

156 # Constant polynomial with no roots 

157 return [] 

158 

159 orig = ctx.prec 

160 tol = +ctx.eps 

161 with ctx.extraprec(extraprec): 

162 deg = len(coeffs) - 1 

163 # Must be monic 

164 lead = ctx.convert(coeffs[0]) 

165 if lead == 1: 

166 coeffs = [ctx.convert(c) for c in coeffs] 

167 else: 

168 coeffs = [c/lead for c in coeffs] 

169 f = lambda x: ctx.polyval(coeffs, x) 

170 if roots_init is None: 

171 roots = [ctx.mpc((0.4+0.9j)**n) for n in xrange(deg)] 

172 else: 

173 roots = [None]*deg; 

174 deg_init = min(deg, len(roots_init)) 

175 roots[:deg_init] = list(roots_init[:deg_init]) 

176 roots[deg_init:] = [ctx.mpc((0.4+0.9j)**n) for n 

177 in xrange(deg_init,deg)] 

178 err = [ctx.one for n in xrange(deg)] 

179 # Durand-Kerner iteration until convergence 

180 for step in xrange(maxsteps): 

181 if abs(max(err)) < tol: 

182 break 

183 for i in xrange(deg): 

184 p = roots[i] 

185 x = f(p) 

186 for j in range(deg): 

187 if i != j: 

188 try: 

189 x /= (p-roots[j]) 

190 except ZeroDivisionError: 

191 continue 

192 roots[i] = p - x 

193 err[i] = abs(x) 

194 if abs(max(err)) >= tol: 

195 raise ctx.NoConvergence("Didn't converge in maxsteps=%d steps." \ 

196 % maxsteps) 

197 # Remove small real or imaginary parts 

198 if cleanup: 

199 for i in xrange(deg): 

200 if abs(roots[i]) < tol: 

201 roots[i] = ctx.zero 

202 elif abs(ctx._im(roots[i])) < tol: 

203 roots[i] = roots[i].real 

204 elif abs(ctx._re(roots[i])) < tol: 

205 roots[i] = roots[i].imag * 1j 

206 roots.sort(key=lambda x: (abs(ctx._im(x)), ctx._re(x))) 

207 if error: 

208 err = max(err) 

209 err = max(err, ctx.ldexp(1, -orig+1)) 

210 return [+r for r in roots], +err 

211 else: 

212 return [+r for r in roots]
