14def egyptian_fraction(r, algorithm="Greedy"): 

15 """ 

16 Return the list of denominators of an Egyptian fraction 

17 expansion [1]_ of the said rational `r`. 

18 

19 Parameters 

20 ========== 

21 

22 r : Rational 

23 a positive rational number. 

24 algorithm : { "Greedy", "Graham Jewett", "Takenouchi", "Golomb" }, optional 

25 Denotes the algorithm to be used (the default is "Greedy"). 

26 

27 Examples 

28 ======== 

29 

30 >>> from sympy import Rational 

31 >>> from sympy.ntheory.egyptian_fraction import egyptian_fraction 

32 >>> egyptian_fraction(Rational(3, 7)) 

33 [3, 11, 231] 

34 >>> egyptian_fraction(Rational(3, 7), "Graham Jewett") 

35 [7, 8, 9, 56, 57, 72, 3192] 

36 >>> egyptian_fraction(Rational(3, 7), "Takenouchi") 

37 [4, 7, 28] 

38 >>> egyptian_fraction(Rational(3, 7), "Golomb") 

39 [3, 15, 35] 

40 >>> egyptian_fraction(Rational(11, 5), "Golomb") 

41 [1, 2, 3, 4, 9, 234, 1118, 2580] 

42 

43 See Also 

44 ======== 

45 

46 sympy.core.numbers.Rational 

47 

48 Notes 

49 ===== 

50 

51 Currently the following algorithms are supported: 

52 

53 1) Greedy Algorithm 

54 

55 Also called the Fibonacci-Sylvester algorithm [2]_. 

56 At each step, extract the largest unit fraction less 

57 than the target and replace the target with the remainder. 

58 

59 It has some distinct properties: 

60 

61 a) Given `p/q` in lowest terms, generates an expansion of maximum 

62 length `p`. Even as the numerators get large, the number of 

63 terms is seldom more than a handful. 

64 

65 b) Uses minimal memory. 

66 

67 c) The terms can blow up (standard examples of this are 5/121 and 

68 31/311). The denominator is at most squared at each step 

69 (doubly-exponential growth) and typically exhibits 

70 singly-exponential growth. 

71 

72 2) Graham Jewett Algorithm 

73 

74 The algorithm suggested by the result of Graham and Jewett. 

75 Note that this has a tendency to blow up: the length of the 

76 resulting expansion is always ``2**(x/gcd(x, y)) - 1``. See [3]_. 

77 

78 3) Takenouchi Algorithm 

79 

80 The algorithm suggested by Takenouchi (1921). 

81 Differs from the Graham-Jewett algorithm only in the handling 

82 of duplicates. See [3]_. 

83 

84 4) Golomb's Algorithm 

85 

86 A method given by Golumb (1962), using modular arithmetic and 

87 inverses. It yields the same results as a method using continued 

88 fractions proposed by Bleicher (1972). See [4]_. 

89 

90 If the given rational is greater than or equal to 1, a greedy algorithm 

91 of summing the harmonic sequence 1/1 + 1/2 + 1/3 + ... is used, taking 

92 all the unit fractions of this sequence until adding one more would be 

93 greater than the given number. This list of denominators is prefixed 

94 to the result from the requested algorithm used on the remainder. For 

95 example, if r is 8/3, using the Greedy algorithm, we get [1, 2, 3, 4, 

96 5, 6, 7, 14, 420], where the beginning of the sequence, [1, 2, 3, 4, 5, 

97 6, 7] is part of the harmonic sequence summing to 363/140, leaving a 

98 remainder of 31/420, which yields [14, 420] by the Greedy algorithm. 

99 The result of egyptian_fraction(Rational(8, 3), "Golomb") is [1, 2, 3, 

100 4, 5, 6, 7, 14, 574, 2788, 6460, 11590, 33062, 113820], and so on. 

101 

102 References 

103 ========== 

104 

105 .. [1] https://en.wikipedia.org/wiki/Egyptian_fraction 

106 .. [2] https://en.wikipedia.org/wiki/Greedy_algorithm_for_Egyptian_fractions 

107 .. [3] https://www.ics.uci.edu/~eppstein/numth/egypt/conflict.html 

108 .. [4] http://ami.ektf.hu/uploads/papers/finalpdf/AMI_42_from129to134.pdf 

109 

110 """ 

111 

112 if r <= 0: 

113 raise ValueError("Value must be positive") 

114 

115 prefix, rem = egypt_harmonic(r) 

116 if rem == 0: 

117 return prefix 

118 x, y = rem.as_numer_denom() 

119 

120 if algorithm == "Greedy": 

121 return prefix + egypt_greedy(x, y) 

122 elif algorithm == "Graham Jewett": 

123 return prefix + egypt_graham_jewett(x, y) 

124 elif algorithm == "Takenouchi": 

125 return prefix + egypt_takenouchi(x, y) 

126 elif algorithm == "Golomb": 

127 return prefix + egypt_golomb(x, y) 

128 else: 

129 raise ValueError("Entered invalid algorithm")

132def egypt_greedy(x, y): 

133 if x == 1: 

134 return [y] 

135 else: 

136 a = (-y) % (x) 

137 b = y*(y//x + 1) 

138 c = gcd(a, b) 

139 if c > 1: 

140 num, denom = a//c, b//c 

141 else: 

142 num, denom = a, b 

143 return [y//x + 1] + egypt_greedy(num, denom)

146def egypt_graham_jewett(x, y): 

147 l = [y] * x 

148 

149 # l is now a list of integers whose reciprocals sum to x/y. 

150 # we shall now proceed to manipulate the elements of l without 

151 # changing the reciprocated sum until all elements are unique. 

152 

153 while len(l) != len(set(l)): 

154 l.sort() # so the list has duplicates. find a smallest pair 

155 for i in range(len(l) - 1): 

156 if l[i] == l[i + 1]: 

157 break 

158 # we have now identified a pair of identical 

159 # elements: l[i] and l[i + 1]. 

160 # now comes the application of the result of graham and jewett: 

161 l[i + 1] = l[i] + 1 

162 # and we just iterate that until the list has no duplicates. 

163 l.append(l[i]*(l[i] + 1)) 

164 return sorted(l)

167def egypt_takenouchi(x, y): 

168 l = [y] * x 

169 while len(l) != len(set(l)): 

170 l.sort() 

171 for i in range(len(l) - 1): 

172 if l[i] == l[i + 1]: 

173 break 

174 k = l[i] 

175 if k % 2 == 0: 

176 l[i] = l[i] // 2 

177 del l[i + 1] 

178 else: 

179 l[i], l[i + 1] = (k + 1)//2, k*(k + 1)//2 

180 return sorted(l)

183def egypt_golomb(x, y): 

184 if x == 1: 

185 return [y] 

186 xp = sympy.polys.ZZ.invert(int(x), int(y)) 

187 rv = [Integer(xp*y)] 

188 rv.extend(egypt_golomb((x*xp - 1)//y, xp)) 

189 return sorted(rv)

192def egypt_harmonic(r): 

193 rv = [] 

194 d = Integer(1) 

195 acc = Integer(0) 

196 while acc + 1/d <= r: 

197 acc += 1/d 

198 rv.append(d) 

199 d += 1 

200 return (rv, r - acc)
