15def giant_steps(start, target, n=2): 

16 """ 

17 Return a list of integers ~= 

18 

19 [start, n*start, ..., target/n^2, target/n, target] 

20 

21 but conservatively rounded so that the quotient between two 

22 successive elements is actually slightly less than n. 

23 

24 With n = 2, this describes suitable precision steps for a 

25 quadratically convergent algorithm such as Newton's method; 

26 with n = 3 steps for cubic convergence (Halley's method), etc. 

27 

28 >>> giant_steps(50,1000) 

29 [66, 128, 253, 502, 1000] 

30 >>> giant_steps(50,1000,4) 

31 [65, 252, 1000] 

32 

33 """ 

34 L = [target] 

35 while L[-1] > start*n: 

36 L = L + [L[-1]//n + 2] 

37 return L[::-1]

141def numeral_python(n, base=10, size=0, digits=stddigits): 

142 """Represent the integer n as a string of digits in the given base. 

143 Recursive division is used to make this function about 3x faster 

144 than Python's str() for converting integers to decimal strings. 

145 

146 The 'size' parameters specifies the number of digits in n; this 

147 number is only used to determine splitting points and need not be 

148 exact.""" 

149 if n <= 0: 

150 if not n: 

151 return "0" 

152 return "-" + numeral(-n, base, size, digits) 

153 # Fast enough to do directly 

154 if size < 250: 

155 return small_numeral(n, base, digits) 

156 # Divide in half 

157 half = (size // 2) + (size & 1) 

158 A, B = divmod(n, base**half) 

159 ad = numeral(A, base, half, digits) 

160 bd = numeral(B, base, half, digits).rjust(half, "0") 

161 return ad + bd

197def isqrt_small_python(x): 

198 """ 

199 Correctly (floor) rounded integer square root, using 

200 division. Fast up to ~200 digits. 

201 """ 

202 if not x: 

203 return x 

204 if x < _1_800: 

205 # Exact with IEEE double precision arithmetic 

206 if x < _1_50: 

207 return int(x**0.5) 

208 # Initial estimate can be any integer >= the true root; round up 

209 r = int(x**0.5 * 1.00000000000001) + 1 

210 else: 

211 bc = bitcount(x) 

212 n = bc//2 

213 r = int((x>>(2*n-100))**0.5+2)<<(n-50) # +2 is to round up 

214 # The following iteration now precisely computes floor(sqrt(x)) 

215 # See e.g. Crandall & Pomerance, "Prime Numbers: A Computational 

216 # Perspective" 

217 while 1: 

218 y = (r+x//r)>>1 

219 if y >= r: 

220 return r 

221 r = y

223def isqrt_fast_python(x): 

224 """ 

225 Fast approximate integer square root, computed using division-free 

226 Newton iteration for large x. For random integers the result is almost 

227 always correct (floor(sqrt(x))), but is 1 ulp too small with a roughly 

228 0.1% probability. If x is very close to an exact square, the answer is 

229 1 ulp wrong with high probability. 

230 

231 With 0 guard bits, the largest error over a set of 10^5 random 

232 inputs of size 1-10^5 bits was 3 ulp. The use of 10 guard bits 

233 almost certainly guarantees a max 1 ulp error. 

234 """ 

235 # Use direct division-based iteration if sqrt(x) < 2^400 

236 # Assume floating-point square root accurate to within 1 ulp, then: 

237 # 0 Newton iterations good to 52 bits 

238 # 1 Newton iterations good to 104 bits 

239 # 2 Newton iterations good to 208 bits 

240 # 3 Newton iterations good to 416 bits 

241 if x < _1_800: 

242 y = int(x**0.5) 

243 if x >= _1_100: 

244 y = (y + x//y) >> 1 

245 if x >= _1_200: 

246 y = (y + x//y) >> 1 

247 if x >= _1_400: 

248 y = (y + x//y) >> 1 

249 return y 

250 bc = bitcount(x) 

251 guard_bits = 10 

252 x <<= 2*guard_bits 

253 bc += 2*guard_bits 

254 bc += (bc&1) 

255 hbc = bc//2 

256 startprec = min(50, hbc) 

257 # Newton iteration for 1/sqrt(x), with floating-point starting value 

258 r = int(2.0**(2*startprec) * (x >> (bc-2*startprec)) ** -0.5) 

259 pp = startprec 

260 for p in giant_steps(startprec, hbc): 

261 # r**2, scaled from real size 2**(-bc) to 2**p 

262 r2 = (r*r) >> (2*pp - p) 

263 # x*r**2, scaled from real size ~1.0 to 2**p 

264 xr2 = ((x >> (bc-p)) * r2) >> p 

265 # New value of r, scaled from real size 2**(-bc/2) to 2**p 

266 r = (r * ((3<<p) - xr2)) >> (pp+1) 

267 pp = p 

268 # (1/sqrt(x))*x = sqrt(x) 

269 return (r*(x>>hbc)) >> (p+guard_bits)

272 """Correctly rounded integer (floor) square root with remainder.""" 

273 # to check cutoff:

318def ifib(n, _cache={}): 

319 """Computes the nth Fibonacci number as an integer, for 

320 integer n.""" 

321 if n < 0: 

322 return (-1)**(-n+1) * ifib(-n) 

323 if n in _cache: 

324 return _cache[n] 

325 m = n 

326 # Use Dijkstra's logarithmic algorithm 

327 # The following implementation is basically equivalent to 

328 # http://en.literateprograms.org/Fibonacci_numbers_(Scheme) 

329 a, b, p, q = MPZ_ONE, MPZ_ZERO, MPZ_ZERO, MPZ_ONE 

330 while n: 

331 if n & 1: 

332 aq = a*q 

333 a, b = b*q+aq+a*p, b*p+aq 

334 n -= 1 

335 else: 

336 qq = q*q 

337 p, q = p*p+qq, qq+2*p*q 

338 n >>= 1 

339 if m < 250: 

340 _cache[m] = b 

341 return b

345def ifac(n, memo={0:1, 1:1}): 

346 """Return n factorial (for integers n >= 0 only).""" 

347 f = memo.get(n) 

348 if f: 

349 return f 

350 k = len(memo) 

351 p = memo[k-1] 

352 MAX = MAX_FACTORIAL_CACHE 

353 while k <= n: 

354 p *= k 

355 if k <= MAX: 

356 memo[k] = p 

357 k += 1 

358 return p

401def isprime(n): 

402 """ 

403 Determines whether n is a prime number. A probabilistic test is 

404 performed if n is very large. No special trick is used for detecting 

405 perfect powers. 

406 

407 >>> sum(list_primes(100000)) 

408 454396537 

409 >>> sum(n*isprime(n) for n in range(100000)) 

410 454396537 

411 

412 """ 

413 n = int(n) 

414 if not n & 1: 

415 return n == 2 

416 if n < 50: 

417 return n in small_odd_primes_set 

418 for p in small_odd_primes: 

419 if not n % p: 

420 return False 

421 m = n-1 

422 s = trailing(m) 

423 d = m >> s 

424 def test(a): 

425 x = pow(a,d,n) 

426 if x == 1 or x == m: 

427 return True 

428 for r in xrange(1,s): 

429 x = x**2 % n 

430 if x == m: 

431 return True 

432 return False 

433 # See http://primes.utm.edu/prove/prove2_3.html 

434 if n < 1373653: 

435 witnesses = [2,3] 

436 elif n < 341550071728321: 

437 witnesses = [2,3,5,7,11,13,17] 

438 else: 

439 witnesses = small_odd_primes 

440 for a in witnesses: 

441 if not test(a): 

442 return False 

443 return True

445def moebius(n): 

446 """ 

447 Evaluates the Moebius function which is `mu(n) = (-1)^k` if `n` 

448 is a product of `k` distinct primes and `mu(n) = 0` otherwise. 

449 

450 TODO: speed up using factorization 

451 """ 

452 n = abs(int(n)) 

453 if n < 2: 

454 return n

456 for p in xrange(2, n+1): 

457 if not (n % p): 

458 if not (n % p**2): 

459 return 0 

460 if not sum(p % f for f in factors): 

461 factors.append(p) 

462 return (-1)**len(factors)

475# Comment by Juan Arias de Reyna: 

476# 

477# I learn this method to compute EulerE[2n] from van de Lune. 

478# 

479# We apply the formula EulerE[2n] = (-1)^n 2**(-2n) sum_{j=0}^n a(2n,2j+1) 

480# 

481# where the numbers a(n,j) vanish for j > n+1 or j <= -1 and satisfies 

482# 

483# a(0,-1) = a(0,0) = 0; a(0,1)= 1; a(0,2) = a(0,3) = 0 

484# 

485# a(n,j) = a(n-1,j) when n+j is even 

486# a(n,j) = (j-1) a(n-1,j-1) + (j+1) a(n-1,j+1) when n+j is odd 

487# 

488# 

489# But we can use only one array unidimensional a(j) since to compute 

490# a(n,j) we only need to know a(n-1,k) where k and j are of different parity 

491# and we have not to conserve the used values. 

492# 

493# We cached up the values of Euler numbers to sufficiently high order. 

494# 

495# Important Observation: If we pretend to use the numbers 

496# EulerE[1], EulerE[2], ... , EulerE[n] 

497# it is convenient to compute first EulerE[n], since the algorithm 

498# computes first all 

499# the previous ones, and keeps them in the CACHE

503def eulernum(m, _cache={0:MPZ_ONE}): 

504 r""" 

505 Computes the Euler numbers `E(n)`, which can be defined as 

506 coefficients of the Taylor expansion of `1/cosh x`: 

507 

508 .. math :: 

509 

510 \frac{1}{\cosh x} = \sum_{n=0}^\infty \frac{E_n}{n!} x^n 

511 

512 Example:: 

513 

514 >>> [int(eulernum(n)) for n in range(11)] 

515 [1, 0, -1, 0, 5, 0, -61, 0, 1385, 0, -50521] 

516 >>> [int(eulernum(n)) for n in range(11)] # test cache 

517 [1, 0, -1, 0, 5, 0, -61, 0, 1385, 0, -50521] 

518 

519 """ 

520 # for odd m > 1, the Euler numbers are zero 

521 if m & 1: 

522 return MPZ_ZERO 

523 f = _cache.get(m) 

524 if f: 

525 return f 

526 MAX = MAX_EULER_CACHE 

527 n = m 

528 a = [MPZ(_) for _ in [0,0,1,0,0,0]] 

529 for n in range(1, m+1): 

530 for j in range(n+1, -1, -2): 

531 a[j+1] = (j-1)*a[j] + (j+1)*a[j+2] 

532 a.append(0) 

533 suma = 0 

534 for k in range(n+1, -1, -2): 

535 suma += a[k+1] 

536 if n <= MAX: 

537 _cache[n] = ((-1)**(n//2))*(suma // 2**n) 

538 if n == m: 

539 return ((-1)**(n//2))*suma // 2**n

541def stirling1(n, k): 

542 """ 

543 Stirling number of the first kind. 

544 """ 

545 if n < 0 or k < 0: 

546 raise ValueError 

547 if k >= n: 

548 return MPZ(n == k) 

549 if k < 1: 

550 return MPZ_ZERO 

551 L = [MPZ_ZERO] * (k+1) 

552 L[1] = MPZ_ONE 

553 for m in xrange(2, n+1): 

554 for j in xrange(min(k, m), 0, -1): 

555 L[j] = (m-1) * L[j] + L[j-1] 

556 return (-1)**(n+k) * L[k]

558def stirling2(n, k): 

559 """ 

560 Stirling number of the second kind. 

561 """ 

562 if n < 0 or k < 0: 

563 raise ValueError 

564 if k >= n: 

565 return MPZ(n == k) 

566 if k <= 1: 

567 return MPZ(k == 1) 

568 s = MPZ_ZERO 

569 t = MPZ_ONE 

570 for j in xrange(k+1): 

571 if (k + j) & 1: 

572 s -= t * MPZ(j)**n 

573 else: 

574 s += t * MPZ(j)**n 

575 t = t * (k - j) // (j + 1) 

576 return s // ifac(k)
