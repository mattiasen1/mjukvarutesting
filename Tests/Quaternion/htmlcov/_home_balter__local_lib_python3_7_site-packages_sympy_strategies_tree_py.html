10def treeapply(tree, join, leaf=identity): 

11 """ Apply functions onto recursive containers (tree) 

12 

13 join - a dictionary mapping container types to functions 

14 e.g. ``{list: minimize, tuple: chain}`` 

15 

16 Keys are containers/iterables. Values are functions [a] -> a. 

17 

18 Examples 

19 ======== 

20 

21 >>> from sympy.strategies.tree import treeapply 

22 >>> tree = [(3, 2), (4, 1)] 

23 >>> treeapply(tree, {list: max, tuple: min}) 

24 2 

25 

26 >>> add = lambda *args: sum(args) 

27 >>> def mul(*args): 

28 ... total = 1 

29 ... for arg in args: 

30 ... total *= arg 

31 ... return total 

32 >>> treeapply(tree, {list: mul, tuple: add}) 

33 25 

34 """ 

35 for typ in join: 

36 if isinstance(tree, typ): 

37 return join[typ](*map(partial(treeapply, join=join, leaf=leaf), 

38 tree)) 

39 return leaf(tree)

41def greedy(tree, objective=identity, **kwargs): 

42 """ Execute a strategic tree. Select alternatives greedily 

43 

44 Trees 

45 ----- 

46 

47 Nodes in a tree can be either 

48 

49 function - a leaf 

50 list - a selection among operations 

51 tuple - a sequence of chained operations 

52 

53 Textual examples 

54 ---------------- 

55 

56 Text: Run f, then run g, e.g. ``lambda x: g(f(x))`` 

57 Code: ``(f, g)`` 

58 

59 Text: Run either f or g, whichever minimizes the objective 

60 Code: ``[f, g]`` 

61 

62 Textx: Run either f or g, whichever is better, then run h 

63 Code: ``([f, g], h)`` 

64 

65 Text: Either expand then simplify or try factor then foosimp. Finally print 

66 Code: ``([(expand, simplify), (factor, foosimp)], print)`` 

67 

68 Objective 

69 --------- 

70 

71 "Better" is determined by the objective keyword. This function makes 

72 choices to minimize the objective. It defaults to the identity. 

73 

74 Examples 

75 ======== 

76 

77 >>> from sympy.strategies.tree import greedy 

78 >>> inc = lambda x: x + 1 

79 >>> dec = lambda x: x - 1 

80 >>> double = lambda x: 2*x 

81 

82 >>> tree = [inc, (dec, double)] # either inc or dec-then-double 

83 >>> fn = greedy(tree) 

84 >>> fn(4) # lowest value comes from the inc 

85 5 

86 >>> fn(1) # lowest value comes from dec then double 

87 0 

88 

89 This function selects between options in a tuple. The result is chosen that 

90 minimizes the objective function. 

91 

92 >>> fn = greedy(tree, objective=lambda x: -x) # maximize 

93 >>> fn(4) # highest value comes from the dec then double 

94 6 

95 >>> fn(1) # highest value comes from the inc 

96 2 

97 

98 Greediness 

99 ---------- 

100 

101 This is a greedy algorithm. In the example: 

102 

103 ([a, b], c) # do either a or b, then do c 

104 

105 the choice between running ``a`` or ``b`` is made without foresight to c 

106 """ 

107 optimize = partial(minimize, objective=objective) 

108 return treeapply(tree, {list: optimize, tuple: chain}, **kwargs)

110def allresults(tree, leaf=yieldify): 

111 """ Execute a strategic tree. Return all possibilities. 

112 

113 Returns a lazy iterator of all possible results 

114 

115 Exhaustiveness 

116 -------------- 

117 

118 This is an exhaustive algorithm. In the example 

119 

120 ([a, b], [c, d]) 

121 

122 All of the results from 

123 

124 (a, c), (b, c), (a, d), (b, d) 

125 

126 are returned. This can lead to combinatorial blowup. 

127 

128 See sympy.strategies.greedy for details on input 

129 """ 

130 return treeapply(tree, {list: branch.multiplex, tuple: branch.chain}, 

131 leaf=leaf)
