17def _preprocess(expr, func=None, hint='_Integral'): 

18 """Prepare expr for solving by making sure that differentiation 

19 is done so that only func remains in unevaluated derivatives and 

20 (if hint doesn't end with _Integral) that doit is applied to all 

21 other derivatives. If hint is None, don't do any differentiation. 

22 (Currently this may cause some simple differential equations to 

23 fail.) 

24 

25 In case func is None, an attempt will be made to autodetect the 

26 function to be solved for. 

27 

28 >>> from sympy.solvers.deutils import _preprocess 

29 >>> from sympy import Derivative, Function, Integral, sin 

30 >>> from sympy.abc import x, y, z 

31 >>> f, g = map(Function, 'fg') 

32 

33 Apply doit to derivatives that contain more than the function 

34 of interest: 

35 

36 >>> _preprocess(Derivative(f(x) + x, x)) 

37 (Derivative(f(x), x) + 1, f(x)) 

38 

39 Do others if the differentiation variable(s) intersect with those 

40 of the function of interest or contain the function of interest: 

41 

42 >>> _preprocess(Derivative(g(x), y, z), f(y)) 

43 (0, f(y)) 

44 >>> _preprocess(Derivative(f(y), z), f(y)) 

45 (0, f(y)) 

46 

47 Do others if the hint doesn't end in '_Integral' (the default 

48 assumes that it does): 

49 

50 >>> _preprocess(Derivative(g(x), y), f(x)) 

51 (Derivative(g(x), y), f(x)) 

52 >>> _preprocess(Derivative(f(x), y), f(x), hint='') 

53 (0, f(x)) 

54 

55 Don't do any derivatives if hint is None: 

56 

57 >>> eq = Derivative(f(x) + 1, x) + Derivative(f(x), y) 

58 >>> _preprocess(eq, f(x), hint=None) 

59 (Derivative(f(x) + 1, x) + Derivative(f(x), y), f(x)) 

60 

61 If it's not clear what the function of interest is, it must be given: 

62 

63 >>> eq = Derivative(f(x) + g(x), x) 

64 >>> _preprocess(eq, g(x)) 

65 (Derivative(f(x), x) + Derivative(g(x), x), g(x)) 

66 >>> try: _preprocess(eq) 

67 ... except ValueError: print("A ValueError was raised.") 

68 A ValueError was raised. 

69 

70 """ 

71 

72 derivs = expr.atoms(Derivative) 

73 if not func: 

74 funcs = set().union(*[d.atoms(AppliedUndef) for d in derivs]) 

75 if len(funcs) != 1: 

76 raise ValueError('The function cannot be ' 

77 'automatically detected for %s.' % expr) 

78 func = funcs.pop() 

79 fvars = set(func.args) 

80 if hint is None: 

81 return expr, func 

82 reps = [(d, d.doit()) for d in derivs if not hint.endswith('_Integral') or 

83 d.has(func) or set(d.variables) & fvars] 

84 eq = expr.subs(reps) 

85 return eq, func

87def ode_order(expr, func): 

88 """ 

89 Returns the order of a given differential 

90 equation with respect to func. 

91 

92 This function is implemented recursively. 

93 

94 Examples 

95 ======== 

96 

97 >>> from sympy import Function 

98 >>> from sympy.solvers.deutils import ode_order 

99 >>> from sympy.abc import x 

100 >>> f, g = map(Function, ['f', 'g']) 

101 >>> ode_order(f(x).diff(x, 2) + f(x).diff(x)**2 + 

102 ... f(x).diff(x), f(x)) 

103 2 

104 >>> ode_order(f(x).diff(x, 2) + g(x).diff(x, 3), f(x)) 

105 2 

106 >>> ode_order(f(x).diff(x, 2) + g(x).diff(x, 3), g(x)) 

107 3 

108 

109 """ 

110 a = Wild('a', exclude=[func]) 

111 if expr.match(a): 

112 return 0 

113 

114 if isinstance(expr, Derivative): 

115 if expr.args[0] == func: 

116 return len(expr.variables) 

117 else: 

118 order = 0 

119 for arg in expr.args[0].args: 

120 order = max(order, ode_order(arg, func) + len(expr.variables)) 

121 return order 

122 else: 

123 order = 0 

124 for arg in expr.args: 

125 order = max(order, ode_order(arg, func)) 

126 return order

128def _desolve(eq, func=None, hint="default", ics=None, simplify=True, **kwargs): 

129 """This is a helper function to dsolve and pdsolve in the ode 

130 and pde modules. 

131 

132 If the hint provided to the function is "default", then a dict with 

133 the following keys are returned 

134 

135 'func' - It provides the function for which the differential equation 

136 has to be solved. This is useful when the expression has 

137 more than one function in it. 

138 

139 'default' - The default key as returned by classifier functions in ode 

140 and pde.py 

141 

142 'hint' - The hint given by the user for which the differential equation 

143 is to be solved. If the hint given by the user is 'default', 

144 then the value of 'hint' and 'default' is the same. 

145 

146 'order' - The order of the function as returned by ode_order 

147 

148 'match' - It returns the match as given by the classifier functions, for 

149 the default hint. 

150 

151 If the hint provided to the function is not "default" and is not in 

152 ('all', 'all_Integral', 'best'), then a dict with the above mentioned keys 

153 is returned along with the keys which are returned when dict in 

154 classify_ode or classify_pde is set True 

155 

156 If the hint given is in ('all', 'all_Integral', 'best'), then this function 

157 returns a nested dict, with the keys, being the set of classified hints 

158 returned by classifier functions, and the values being the dict of form 

159 as mentioned above. 

160 

161 Key 'eq' is a common key to all the above mentioned hints which returns an 

162 expression if eq given by user is an Equality. 

163 

164 See Also 

165 ======== 

166 classify_ode(ode.py) 

167 classify_pde(pde.py) 

168 """ 

169 prep = kwargs.pop('prep', True) 

170 if isinstance(eq, Equality): 

171 eq = eq.lhs - eq.rhs 

172 

173 # preprocess the equation and find func if not given 

174 if prep or func is None: 

175 eq, func = _preprocess(eq, func) 

176 prep = False 

177 

178 # type is an argument passed by the solve functions in ode and pde.py 

179 # that identifies whether the function caller is an ordinary 

180 # or partial differential equation. Accordingly corresponding 

181 # changes are made in the function. 

182 type = kwargs.get('type', None) 

183 xi = kwargs.get('xi') 

184 eta = kwargs.get('eta') 

185 x0 = kwargs.get('x0', 0) 

186 terms = kwargs.get('n') 

187 

188 if type == 'ode': 

189 from sympy.solvers.ode import classify_ode, allhints 

190 classifier = classify_ode 

191 string = 'ODE ' 

192 dummy = '' 

193 

194 elif type == 'pde': 

195 from sympy.solvers.pde import classify_pde, allhints 

196 classifier = classify_pde 

197 string = 'PDE ' 

198 dummy = 'p' 

199 

200 # Magic that should only be used internally. Prevents classify_ode from 

201 # being called more than it needs to be by passing its results through 

202 # recursive calls. 

203 if kwargs.get('classify', True): 

204 hints = classifier(eq, func, dict=True, ics=ics, xi=xi, eta=eta, 

205 n=terms, x0=x0, prep=prep) 

206 

207 else: 

208 # Here is what all this means: 

209 # 

210 # hint: The hint method given to _desolve() by the user. 

211 # hints: The dictionary of hints that match the DE, along with other 

212 # information (including the internal pass-through magic). 

213 # default: The default hint to return, the first hint from allhints 

214 # that matches the hint; obtained from classify_ode(). 

215 # match: Dictionary containing the match dictionary for each hint 

216 # (the parts of the DE for solving). When going through the 

217 # hints in "all", this holds the match string for the current 

218 # hint. 

219 # order: The order of the DE, as determined by ode_order(). 

220 hints = kwargs.get('hint', 

221 {'default': hint, 

222 hint: kwargs['match'], 

223 'order': kwargs['order']}) 

224 if hints['order'] == 0: 

225 raise ValueError( 

226 str(eq) + " is not a differential equation in " + str(func)) 

227 

228 if not hints['default']: 

229 # classify_ode will set hints['default'] to None if no hints match. 

230 if hint not in allhints and hint != 'default': 

231 raise ValueError("Hint not recognized: " + hint) 

232 elif hint not in hints['ordered_hints'] and hint != 'default': 

233 raise ValueError(string + str(eq) + " does not match hint " + hint) 

234 else: 

235 raise NotImplementedError(dummy + "solve" + ": Cannot solve " + str(eq)) 

236 if hint == 'default': 

237 return _desolve(eq, func, ics=ics, hint=hints['default'], simplify=simplify, 

238 prep=prep, x0=x0, classify=False, order=hints['order'], 

239 match=hints[hints['default']], xi=xi, eta=eta, n=terms, type=type) 

240 elif hint in ('all', 'all_Integral', 'best'): 

241 retdict = {} 

242 failedhints = {} 

243 gethints = set(hints) - set(['order', 'default', 'ordered_hints']) 

244 if hint == 'all_Integral': 

245 for i in hints: 

246 if i.endswith('_Integral'): 

247 gethints.remove(i[:-len('_Integral')]) 

248 # special cases 

249 for k in ["1st_homogeneous_coeff_best", "1st_power_series", 

250 "lie_group", "2nd_power_series_ordinary", "2nd_power_series_regular"]: 

251 if k in gethints: 

252 gethints.remove(k) 

253 for i in gethints: 

254 sol = _desolve(eq, func, ics=ics, hint=i, x0=x0, simplify=simplify, prep=prep, 

255 classify=False, n=terms, order=hints['order'], match=hints[i], type=type) 

256 retdict[i] = sol 

257 retdict['all'] = True 

258 retdict['eq'] = eq 

259 return retdict 

260 elif hint not in allhints: # and hint not in ('default', 'ordered_hints'): 

261 raise ValueError("Hint not recognized: " + hint) 

262 elif hint not in hints: 

263 raise ValueError(string + str(eq) + " does not match hint " + hint) 

264 else: 

265 # Key added to identify the hint needed to solve the equation 

266 hints['hint'] = hint 

267 hints.update({'func': func, 'eq': eq}) 

268 return hints
