235 # get one more digit so the extra testing below can be avoided 

236 # unless the rounded value rounded to an integer, e.g. if an

240 # digit we know that 1.9 is not the same as 1 and there is no

520 def is_constant(self, *wrt, **flags): 

521 """Return True if self is constant, False if not, or None if 

522 the constancy could not be determined conclusively. 

523 

524 If an expression has no free symbols then it is a constant. If 

525 there are free symbols it is possible that the expression is a 

526 constant, perhaps (but not necessarily) zero. To test such 

527 expressions, two strategies are tried: 

528 

529 1) numerical evaluation at two random points. If two such evaluations 

530 give two different values and the values have a precision greater than 

531 1 then self is not constant. If the evaluations agree or could not be 

532 obtained with any precision, no decision is made. The numerical testing 

533 is done only if ``wrt`` is different than the free symbols. 

534 

535 2) differentiation with respect to variables in 'wrt' (or all free 

536 symbols if omitted) to see if the expression is constant or not. This 

537 will not always lead to an expression that is zero even though an 

538 expression is constant (see added test in test_expr.py). If 

539 all derivatives are zero then self is constant with respect to the 

540 given symbols. 

541 

542 If neither evaluation nor differentiation can prove the expression is 

543 constant, None is returned unless two numerical values happened to be 

544 the same and the flag ``failing_number`` is True -- in that case the 

545 numerical value will be returned. 

546 

547 If flag simplify=False is passed, self will not be simplified; 

548 the default is True since self should be simplified before testing. 

549 

550 Examples 

551 ======== 

552 

553 >>> from sympy import cos, sin, Sum, S, pi 

554 >>> from sympy.abc import a, n, x, y 

555 >>> x.is_constant() 

556 False 

557 >>> S(2).is_constant() 

558 True 

559 >>> Sum(x, (x, 1, 10)).is_constant() 

560 True 

561 >>> Sum(x, (x, 1, n)).is_constant() 

562 False 

563 >>> Sum(x, (x, 1, n)).is_constant(y) 

564 True 

565 >>> Sum(x, (x, 1, n)).is_constant(n) 

566 False 

567 >>> Sum(x, (x, 1, n)).is_constant(x) 

568 True 

569 >>> eq = a*cos(x)**2 + a*sin(x)**2 - a 

570 >>> eq.is_constant() 

571 True 

572 >>> eq.subs({x: pi, a: 2}) == eq.subs({x: pi, a: 3}) == 0 

573 True 

574 

575 >>> (0**x).is_constant() 

576 False 

577 >>> x.is_constant() 

578 False 

579 >>> (x**x).is_constant() 

580 False 

581 >>> one = cos(x)**2 + sin(x)**2

597 # free symbols then this expression is constant wrt those symbols

602

610 # shouldn't, but hopefully it will never give True unless it is sure. 

611 if expr.is_zero:

617 # try 0 (for a) and 1 (for b) 

618 try:

630 if b is S.NaN:

640 return False

643 if b is not None and b is not S.NaN: 

644 if b.equals(a) is False:

666 def equals(self, other, failing_expression=False): 

667 """Return True if self == other, False if it doesn't, or None. If 

668 failing_expression is True then the expression which did not simplify 

669 to a 0 will be returned instead of None. 

670 

671 If ``self`` is a Number (or complex number) that is not zero, then 

672 the result is False. 

673 

674 If ``self`` is a number and has not evaluated to zero, evalf will be 

675 used to test whether the expression evaluates to zero. If it does so 

676 and the result has significance (i.e. the precision is either -1, for 

677 a Rational result, or is greater than 1) then the evalf value will be 

678 used to return True or False. 

679 

680 """ 

681 from sympy.simplify.simplify import nsimplify, simplify 

682 from sympy.solvers.solveset import solveset 

683 from sympy.polys.polyerrors import NotAlgebraic 

684 from sympy.polys.numberfields import minimal_polynomial 

685 

686 other = sympify(other) 

687 if self == other: 

688 return True 

689 

690 # they aren't the same so see if we can make the difference 0; 

691 # don't worry about doing simplification steps one at a time 

692 # because if the expression ever goes to 0 then the subsequent 

693 # simplification steps that are done will be very fast. 

694 diff = factor_terms(simplify(self - other), radical=True) 

695 

696 if not diff: 

697 return True 

698 

699 if not diff.has(Add, Mod): 

700 # if there is no expanding to be done after simplifying 

701 # then this can't be a zero 

702 return False 

703 

704 constant = diff.is_constant(simplify=False, failing_number=True) 

705 

706 if constant is False: 

707 return False 

708 

709 if constant is None and not diff.is_number: 

710 # e.g. unless the right simplification is done, a symbolic 

711 # zero is possible (see expression of issue 6829: without 

712 # simplification constant will be None). 

713 return 

714 

715 if constant is True: 

716 ndiff = diff._random() 

717 if ndiff: 

718 return False 

719 

720 # sometimes we can use a simplified result to give a clue as to 

721 # what the expression should be; if the expression is *not* zero 

722 # then we should have been able to compute that and so now 

723 # we can just consider the cases where the approximation appears 

724 # to be zero -- we try to prove it via minimal_polynomial. 

725 if diff.is_number: 

726 approx = diff.nsimplify() 

727 if not approx: 

728 # try to prove via self-consistency 

729 surds = [s for s in diff.atoms(Pow) if s.args[0].is_Integer] 

730 # it seems to work better to try big ones first 

731 surds.sort(key=lambda x: -x.args[0]) 

732 for s in surds: 

733 try: 

734 # simplify is False here -- this expression has already 

735 # been identified as being hard to identify as zero; 

736 # we will handle the checking ourselves using nsimplify 

737 # to see if we are in the right ballpark or not and if so 

738 # *then* the simplification will be attempted. 

739 if s.is_Symbol: 

740 sol = list(solveset(diff, s)) 

741 else: 

742 sol = [s] 

743 if sol: 

744 if s in sol: 

745 return True 

746 if s.is_real: 

747 if any(nsimplify(si, [s]) == s and simplify(si) == s 

748 for si in sol): 

749 return True 

750 except NotImplementedError: 

751 pass 

752 

753 # try to prove with minimal_polynomial but know when 

754 # *not* to use this or else it can take a long time. e.g. issue 8354 

755 if True: # change True to condition that assures non-hang 

756 try: 

757 mp = minimal_polynomial(diff) 

758 if mp.is_Symbol: 

759 return True 

760 return False 

761 except (NotAlgebraic, NotImplementedError): 

762 pass 

763 

764 # diff has not simplified to zero; constant is either None, True 

765 # or the number with significance (prec != 1) that was randomly 

766 # calculated twice as the same value. 

767 if constant not in (True, None) and constant != 0: 

768 return False 

769 

770 if failing_expression: 

771 return diff 

772 return None

844 def _eval_interval(self, x, a, b): 

845 """ 

846 Returns evaluation over an interval. For most functions this is: 

847 

848 self.subs(x, b) - self.subs(x, a), 

849 

850 possibly using limit() if NaN is returned from subs, or if 

851 singularities are found between a and b. 

852 

853 If b or a is None, it only evaluates -self.subs(x, a) or self.subs(b, x), 

854 respectively. 

855 

856 """ 

857 from sympy.series import limit, Limit 

858 from sympy.solvers.solveset import solveset 

859 from sympy.sets.sets import Interval 

860 from sympy.functions.elementary.exponential import log 

861 from sympy.calculus.util import AccumBounds 

862 

863 if (a is None and b is None): 

864 raise ValueError('Both interval ends cannot be None.') 

865 

866 if a == b: 

867 return 0 

868 

869 if a is None: 

870 A = 0 

871 else: 

872 A = self.subs(x, a) 

873 if A.has(S.NaN, S.Infinity, S.NegativeInfinity, S.ComplexInfinity, AccumBounds): 

874 if (a < b) != False: 

875 A = limit(self, x, a,"+") 

876 else: 

877 A = limit(self, x, a,"-") 

878 

879 if A is S.NaN: 

880 return A 

881 if isinstance(A, Limit): 

882 raise NotImplementedError("Could not compute limit") 

883 

884 if b is None: 

885 B = 0 

886 else: 

887 B = self.subs(x, b) 

888 if B.has(S.NaN, S.Infinity, S.NegativeInfinity, S.ComplexInfinity, AccumBounds): 

889 if (a < b) != False: 

890 B = limit(self, x, b,"-") 

891 else: 

892 B = limit(self, x, b,"+") 

893 

894 if isinstance(B, Limit): 

895 raise NotImplementedError("Could not compute limit") 

896 

897 if (a and b) is None: 

898 return B - A 

899 

900 value = B - A 

901 

902 if a.is_comparable and b.is_comparable: 

903 if a < b: 

904 domain = Interval(a, b) 

905 else: 

906 domain = Interval(b, a) 

907 # check the singularities of self within the interval 

908 # if singularities is a ConditionSet (not iterable), catch the exception and pass 

909 singularities = solveset(self.cancel().as_numer_denom()[1], x, 

910 domain=domain) 

911 for logterm in self.atoms(log): 

912 singularities = singularities | solveset(logterm.args[0], x, 

913 domain=domain) 

914 try: 

915 for s in singularities: 

916 if value is S.NaN: 

917 # no need to keep adding, it will stay NaN 

918 break 

919 if not s.is_comparable: 

920 continue 

921 if (a < s) == (s < b) == True: 

922 value += -limit(self, x, s, "+") + limit(self, x, s, "-") 

923 elif (b < s) == (s < a) == True: 

924 value += limit(self, x, s, "+") - limit(self, x, s, "-") 

925 except TypeError: 

926 pass 

927 

928 return value

1182 def args_cnc(self, cset=False, warn=True, split_1=True): 

1183 """Return [commutative factors, non-commutative factors] of self. 

1184 

1185 self is treated as a Mul and the ordering of the factors is maintained. 

1186 If ``cset`` is True the commutative factors will be returned in a set. 

1187 If there were repeated factors (as may happen with an unevaluated Mul) 

1188 then an error will be raised unless it is explicitly suppressed by 

1189 setting ``warn`` to False. 

1190 

1191 Note: -1 is always separated from a Number unless split_1 is False. 

1192 

1193 >>> from sympy import symbols, oo 

1194 >>> A, B = symbols('A B', commutative=0) 

1195 >>> x, y = symbols('x y') 

1196 >>> (-2*x*y).args_cnc() 

1197 [[-1, 2, x, y], []] 

1198 >>> (-2.5*x).args_cnc() 

1199 [[-1, 2.5, x], []] 

1200 >>> (-2*x*A*B*y).args_cnc() 

1201 [[-1, 2, x, y], [A, B]] 

1202 >>> (-2*x*A*B*y).args_cnc(split_1=False) 

1203 [[-2, x, y], [A, B]] 

1204 >>> (-2*x*y).args_cnc(cset=True) 

1205 [{-1, 2, x, y}, []] 

1206 

1207 The arg is always treated as a Mul: 

1208 

1209 >>> (-2 + x + A).args_cnc() 

1210 [[], [x - 2 + A]] 

1211 >>> (-oo).args_cnc() # -oo is a singleton 

1212 [[-1, oo], []] 

1213 """ 

1214 

1215 if self.is_Mul: 

1216 args = list(self.args) 

1217 else: 

1218 args = [self] 

1219 for i, mi in enumerate(args): 

1220 if not mi.is_commutative: 

1221 c = args[:i] 

1222 nc = args[i:] 

1223 break 

1224 else: 

1225 c = args 

1226 nc = [] 

1227 

1228 if c and split_1 and ( 

1229 c[0].is_Number and 

1230 c[0].is_negative and 

1231 c[0] is not S.NegativeOne): 

1232 c[:1] = [S.NegativeOne, -c[0]] 

1233 

1234 if cset: 

1235 clen = len(c) 

1236 c = set(c) 

1237 if clen and warn and len(c) != clen: 

1238 raise ValueError('repeated commutative arguments: %s' % 

1239 [ci for ci in c if list(self.args).count(ci) > 1]) 

1240 return [c, nc]

1242 def coeff(self, x, n=1, right=False): 

1243 """ 

1244 Returns the coefficient from the term(s) containing ``x**n``. If ``n`` 

1245 is zero then all terms independent of ``x`` will be returned. 

1246 

1247 When ``x`` is noncommutative, the coefficient to the left (default) or 

1248 right of ``x`` can be returned. The keyword 'right' is ignored when 

1249 ``x`` is commutative. 

1250 

1251 See Also 

1252 ======== 

1253 

1254 as_coefficient: separate the expression into a coefficient and factor 

1255 as_coeff_Add: separate the additive constant from an expression 

1256 as_coeff_Mul: separate the multiplicative constant from an expression 

1257 as_independent: separate x-dependent terms/factors from others 

1258 sympy.polys.polytools.coeff_monomial: efficiently find the single coefficient of a monomial in Poly 

1259 sympy.polys.polytools.nth: like coeff_monomial but powers of monomial terms are used 

1260

1262 ========

1265 >>> from sympy.abc import x, y, z 

1266

1268 

1269 >>> (-x + 2*y).coeff(-1) 

1270 x 

1271 >>> (x - 2*y).coeff(-1) 

1272 2*y 

1273

1275 

1276 >>> (x + 2*y).coeff(1) 

1277 x 

1278 >>> (3 + 2*x + 4*x**2).coeff(1) 

1279 0 

1280

1282 expr.as_independent(x)[0] is returned (and 0 will be returned instead 

1283 of None):

1285 >>> (3 + 2*x + 4*x**2).coeff(x, 0) 

1286 3

1288 >>> eq 

1289 x**3 + 3*x**2 + 3*x + 2 

1290 >>> [eq.coeff(x, i) for i in reversed(range(4))] 

1291 [1, 3, 3, 2] 

1292 >>> eq -= 2 

1293 >>> [eq.coeff(x, i) for i in reversed(range(4))] 

1294 [1, 3, 3, 0]

1297

1302 x

1395 def find(l, sub, first=True): 

1396 """ Find where list sub appears in list l. When ``first`` is True 

1397 the first occurrence from the left is returned, else the last 

1398 occurrence is returned. Return None if sub is not in l.

1411 n = len(sub) 

1412 if not first: 

1413 l.reverse() 

1414 sub.reverse() 

1415 for i in range(0, len(l) - n + 1): 

1416 if all(l[i + j] == sub[j] for j in range(n)): 

1417 break 

1418 else: 

1419 i = None 

1420 if not first: 

1421 l.reverse() 

1422 sub.reverse() 

1423 if i is not None and not first:

1425 return i

1427 co = [] 

1428 args = Add.make_args(self) 

1429 self_c = self.is_commutative 

1430 x_c = x.is_commutative

1445 elif co:

1447 elif x_c: 

1448 xargs = x.args_cnc(cset=True, warn=False)[0] 

1449 for a in args:

1451 if len(xargs) > len(margs): 

1452 continue 

1453 resid = margs.difference(xargs) 

1454 if len(resid) + len(xargs) == len(margs): 

1455 co.append(Mul(*(list(resid) + nc))) 

1456 if co == []:

1458 elif co: 

1459 return Add(*co) 

1460 else: # both nc 

1461 xargs, nx = x.args_cnc(cset=True) 

1462 # find the parts that pass the commutative terms 

1463 for a in args:

1465 if len(xargs) > len(margs): 

1466 continue 

1467 resid = margs.difference(xargs) 

1468 if len(resid) + len(xargs) == len(margs): 

1469 co.append((resid, nc)) 

1470 # now check the non-comm parts 

1471 if not co: 

1472 return S.Zero 

1473 if all(n == co[0][1] for r, n in co): 

1474 ii = find(co[0][1], nx, right) 

1475 if ii is not None: 

1476 if not right: 

1477 return Mul(Add(*[Mul(*r) for r, c in co]), Mul(*co[0][1][:ii])) 

1478 else: 

1479 return Mul(*co[0][1][ii + len(nx):]) 

1480 beg = reduce(incommon, (n[1] for n in co)) 

1481 if beg: 

1482 ii = find(beg, nx, right) 

1483 if ii is not None: 

1484 if not right: 

1485 gcdc = co[0][0] 

1486 for i in range(1, len(co)): 

1487 gcdc = gcdc.intersection(co[i][0]) 

1488 if not gcdc: 

1489 break

1491 else:

1493 return Add(*[Mul(*(list(r) + n[m:])) for r, n in co]) 

1494 end = list(reversed( 

1495 reduce(incommon, (list(reversed(n[1])) for n in co)))) 

1496 if end: 

1497 ii = find(end, nx, right) 

1498 if ii is not None: 

1499 if not right: 

1500 return Add(*[Mul(*(list(r) + n[:-len(end) + ii])) for r, n in co]) 

1501 else: 

1502 return Mul(*end[ii + len(nx):]) 

1503 # look for single match 

1504 hit = None 

1505 for i, (r, n) in enumerate(co): 

1506 ii = find(n, nx, right) 

1507 if ii is not None: 

1508 if not hit: 

1509 hit = ii, r, n 

1510 else: 

1511 break 

1512 else: 

1513 if hit: 

1514 ii, r, n = hit 

1515 if not right: 

1516 return Mul(*(list(r) + n[:ii])) 

1517 else: 

1518 return Mul(*n[ii + len(nx):])
