70 def _new_rawargs(self, *args, **kwargs): 

71 """Create new instance of own class with args exactly as provided by 

72 caller but returning the self class identity if args is empty. 

73 

74 This is handy when we want to optimize things, e.g. 

75 

76 >>> from sympy import Mul, S 

77 >>> from sympy.abc import x, y 

78 >>> e = Mul(3, x, y) 

79 >>> e.args 

80 (3, x, y) 

81 >>> Mul(*e.args[1:]) 

82 x*y 

83 >>> e._new_rawargs(*e.args[1:]) # the same as above, but faster 

84 x*y 

85 

86 Note: use this with caution. There is no checking of arguments at 

87 all. This is best used when you are rebuilding an Add or Mul after 

88 simply removing one or more args. If, for example, modifications, 

89 result in extra 1s being inserted (as when collecting an 

90 expression's numerators and denominators) they will not show up in 

91 the result but a Mul will be returned nonetheless: 

92 

93 >>> m = (x*y)._new_rawargs(S.One, x); m 

94 x 

95 >>> m == x 

96 False 

97 >>> m.is_Mul 

98 True 

99 

100 Another issue to be aware of is that the commutativity of the result 

101 is based on the commutativity of self. If you are rebuilding the 

102 terms that came from a commutative object then there will be no 

103 problem, but if self was non-commutative then what you are 

104 rebuilding may now be commutative. 

105 

106 Although this routine tries to do as little as possible with the 

107 input, getting the commutativity right is important, so this level 

108 of safety is enforced: commutativity will always be recomputed if 

109 self is non-commutative and kwarg `reeval=False` has not been 

110 passed. 

111 """ 

112 if kwargs.pop('reeval', True) and self.is_commutative is False: 

113 is_commutative = None 

114 else: 

115 is_commutative = self.is_commutative 

116 return self._from_args(args, is_commutative)

134 def _matches_commutative(self, expr, repl_dict={}, old=False): 

135 """ 

136 Matches Add/Mul "pattern" to an expression "expr". 

137 

138 repl_dict ... a dictionary of (wild: expression) pairs, that get 

139 returned with the results 

140 

141 This function is the main workhorse for Add/Mul. 

142 

143 For instance: 

144 

145 >>> from sympy import symbols, Wild, sin 

146 >>> a = Wild("a") 

147 >>> b = Wild("b") 

148 >>> c = Wild("c") 

149 >>> x, y, z = symbols("x y z") 

150 >>> (a+sin(b)*c)._matches_commutative(x+sin(y)*z) 

151 {a_: x, b_: y, c_: z} 

152 

153 In the example above, "a+sin(b)*c" is the pattern, and "x+sin(y)*z" is 

154 the expression. 

155 

156 The repl_dict contains parts that were already matched. For example 

157 here: 

158 

159 >>> (x+sin(b)*c)._matches_commutative(x+sin(y)*z, repl_dict={a: x}) 

160 {a_: x, b_: y, c_: z} 

161 

162 the only function of the repl_dict is to return it in the 

163 result, e.g. if you omit it: 

164 

165 >>> (x+sin(b)*c)._matches_commutative(x+sin(y)*z) 

166 {b_: y, c_: z} 

167 

168 the "a: x" is not returned in the result, but otherwise it is 

169 equivalent. 

170 

171 """ 

172 # make sure expr is Expr if pattern is Expr 

173 from .expr import Add, Expr 

174 from sympy import Mul 

175 if isinstance(self, Expr) and not isinstance(expr, Expr): 

176 return None 

177 

178 # handle simple patterns 

179 if self == expr: 

180 return repl_dict 

181 

182 d = self._matches_simple(expr, repl_dict) 

183 if d is not None: 

184 return d 

185 

186 # eliminate exact part from pattern: (2+a+w1+w2).matches(expr) -> (w1+w2).matches(expr-a-2) 

187 from .function import WildFunction 

188 from .symbol import Wild 

189 wild_part, exact_part = sift(self.args, lambda p: 

190 p.has(Wild, WildFunction) and not expr.has(p), 

191 binary=True) 

192 if not exact_part: 

193 wild_part = list(ordered(wild_part)) 

194 else: 

195 exact = self._new_rawargs(*exact_part) 

196 free = expr.free_symbols 

197 if free and (exact.free_symbols - free): 

198 # there are symbols in the exact part that are not 

199 # in the expr; but if there are no free symbols, let 

200 # the matching continue 

201 return None 

202 newexpr = self._combine_inverse(expr, exact) 

203 if not old and (expr.is_Add or expr.is_Mul): 

204 if newexpr.count_ops() > expr.count_ops(): 

205 return None 

206 newpattern = self._new_rawargs(*wild_part) 

207 return newpattern.matches(newexpr, repl_dict) 

208 

209 # now to real work ;) 

210 i = 0 

211 saw = set() 

212 while expr not in saw: 

213 saw.add(expr) 

214 expr_list = (self.identity,) + tuple(ordered(self.make_args(expr))) 

215 for last_op in reversed(expr_list): 

216 for w in reversed(wild_part): 

217 d1 = w.matches(last_op, repl_dict) 

218 if d1 is not None: 

219 d2 = self.xreplace(d1).matches(expr, d1) 

220 if d2 is not None: 

221 return d2 

222 

223 if i == 0: 

224 if self.is_Mul: 

225 # make e**i look like Mul 

226 if expr.is_Pow and expr.exp.is_Integer: 

227 if expr.exp > 0: 

228 expr = Mul(*[expr.base, expr.base**(expr.exp - 1)], evaluate=False) 

229 else: 

230 expr = Mul(*[1/expr.base, expr.base**(expr.exp + 1)], evaluate=False) 

231 i += 1 

232 continue 

233 

234 elif self.is_Add: 

235 # make i*e look like Add 

236 c, e = expr.as_coeff_Mul() 

237 if abs(c) > 1: 

238 if c > 0: 

239 expr = Add(*[e, (c - 1)*e], evaluate=False) 

240 else: 

241 expr = Add(*[-e, (c + 1)*e], evaluate=False) 

242 i += 1 

243 continue 

244 

245 # try collection on non-Wild symbols 

246 from sympy.simplify.radsimp import collect 

247 was = expr 

248 did = set() 

249 for w in reversed(wild_part): 

250 c, w = w.as_coeff_mul(Wild) 

251 free = c.free_symbols - did 

252 if free: 

253 did.update(free) 

254 expr = collect(expr, free) 

255 if expr != was: 

256 i += 0 

257 continue 

258 

259 break # if we didn't continue, there is nothing more to do 

260 

261 return

293 def _eval_evalf(self, prec): 

294 """ 

295 Evaluate the parts of self that are numbers; if the whole thing 

296 was a number with no functions it would have been evaluated, but 

297 it wasn't so we must judiciously extract the numbers and reconstruct 

298 the object. This is *not* simply replacing numbers with evaluated 

299 numbers. Nunmbers should be handled in the largest pure-number 

300 expression as possible. So the code below separates ``self`` into 

301 number and non-number parts and evaluates the number parts and 

302 walks the args of the non-number part recursively (doing the same 

303 thing). 

304 """ 

305 from .add import Add 

306 from .mul import Mul 

307 from .symbol import Symbol 

308 from .function import AppliedUndef 

309 if isinstance(self, (Mul, Add)): 

310 x, tail = self.as_independent(Symbol, AppliedUndef) 

311 # if x is an AssocOp Function then the _evalf below will 

312 # call _eval_evalf (here) so we must break the recursion 

313 if not (tail is self.identity or 

314 isinstance(x, AssocOp) and x.is_Function or 

315 x is self.identity and isinstance(tail, AssocOp)): 

316 # here, we have a number so we just call to _evalf with prec; 

317 # prec is not the same as n, it is the binary precision so 

318 # that's why we don't call to evalf. 

319 x = x._evalf(prec) if x is not self.identity else self.identity 

320 args = [] 

321 tail_args = tuple(self.func.make_args(tail)) 

322 for a in tail_args: 

323 # here we call to _eval_evalf since we don't know what we 

324 # are dealing with and all other _eval_evalf routines should 

325 # be doing the same thing (i.e. taking binary prec and 

326 # finding the evalf-able args) 

327 newa = a._eval_evalf(prec) 

328 if newa is None: 

329 args.append(a) 

330 else: 

331 args.append(newa) 

332 return self.func(x, *args) 

333 

334 # this is the same as above, but there were no pure-number args to 

335 # deal with 

336 args = [] 

337 for a in self.args: 

338 newa = a._eval_evalf(prec) 

339 if newa is None: 

340 args.append(a) 

341 else: 

342 args.append(newa) 

343 return self.func(*args)

371class LatticeOp(AssocOp): 

372 """ 

373 Join/meet operations of an algebraic lattice[1]. 

374 

375 These binary operations are associative (op(op(a, b), c) = op(a, op(b, c))), 

376 commutative (op(a, b) = op(b, a)) and idempotent (op(a, a) = op(a) = a). 

377 Common examples are AND, OR, Union, Intersection, max or min. They have an 

378 identity element (op(identity, a) = a) and an absorbing element 

379 conventionally called zero (op(zero, a) = zero). 

380 

381 This is an abstract base class, concrete derived classes must declare 

382 attributes zero and identity. All defining properties are then respected. 

383 

384 >>> from sympy import Integer 

385 >>> from sympy.core.operations import LatticeOp 

386 >>> class my_join(LatticeOp): 

387 ... zero = Integer(0) 

388 ... identity = Integer(1) 

389 >>> my_join(2, 3) == my_join(3, 2) 

390 True 

391 >>> my_join(2, my_join(3, 4)) == my_join(2, 3, 4) 

392 True 

393 >>> my_join(0, 1, 4, 2, 3, 4) 

394 0 

395 >>> my_join(1, 2) 

396 2 

397 

398 References: 

399 

400 [1] - https://en.wikipedia.org/wiki/Lattice_%28order%29 

401 """ 

402 

403 is_commutative = True 

404 

405 def __new__(cls, *args, **options): 

406 args = (_sympify(arg) for arg in args)

408 try: 

409 # /!\ args is a generator and _new_args_filter

411 # is done so short-circuiting can be done 

412 # without having to sympify all values 

413 _args = frozenset(cls._new_args_filter(args)) 

414 except ShortCircuit: 

415 return sympify(cls.zero) 

416 if not _args: 

417 return sympify(cls.identity) 

418 elif len(_args) == 1: 

419 return set(_args).pop() 

420 else: 

421 # XXX in almost every other case for __new__, *_args is 

422 # passed along, but the expectation here is for _args 

423 obj = super(AssocOp, cls).__new__(cls, _args) 

424 obj._argset = _args 

425 return obj

429 """Generator filtering args"""

431 for arg in arg_sequence:
