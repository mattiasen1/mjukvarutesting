309def _get_coeff_exp(expr, x): 

310 """ 

311 When expr is known to be of the form c*x**b, with c and/or b possibly 1, 

312 return c, b. 

313 

314 >>> from sympy.abc import x, a, b 

315 >>> from sympy.integrals.meijerint import _get_coeff_exp 

316 >>> _get_coeff_exp(a*x**b, x) 

317 (a, b) 

318 >>> _get_coeff_exp(x, x) 

319 (1, 1) 

320 >>> _get_coeff_exp(2*x, x) 

321 (2, 1) 

322 >>> _get_coeff_exp(x**3, x) 

323 (1, 3) 

324 """ 

325 from sympy import powsimp 

326 (c, m) = expand_power_base(powsimp(expr)).as_coeff_mul(x) 

327 if not m: 

328 return c, S(0) 

329 [m] = m 

330 if m.is_Pow: 

331 if m.base != x: 

332 raise _CoeffExpValueError('expr not of form a*x**b') 

333 return c, m.exp 

334 elif m == x: 

335 return c, S(1) 

336 else: 

337 raise _CoeffExpValueError('expr not of form a*x**b: %s' % expr)

340def _exponents(expr, x): 

341 """ 

342 Find the exponents of ``x`` (not including zero) in ``expr``. 

343 

344 >>> from sympy.integrals.meijerint import _exponents 

345 >>> from sympy.abc import x, y 

346 >>> from sympy import sin 

347 >>> _exponents(x, x) 

348 {1} 

349 >>> _exponents(x**2, x) 

350 {2} 

351 >>> _exponents(x**2 + x, x) 

352 {1, 2} 

353 >>> _exponents(x**3*sin(x + x**y) + 1/x, x) 

354 {-1, 1, 3, y} 

355 """ 

356 def _exponents_(expr, x, res): 

357 if expr == x: 

358 res.update([1]) 

359 return 

360 if expr.is_Pow and expr.base == x: 

361 res.update([expr.exp]) 

362 return 

363 for arg in expr.args: 

364 _exponents_(arg, x, res) 

365 res = set() 

366 _exponents_(expr, x, res) 

367 return res

376def _find_splitting_points(expr, x): 

377 """ 

378 Find numbers a such that a linear substitution x -> x + a would 

379 (hopefully) simplify expr. 

380 

381 >>> from sympy.integrals.meijerint import _find_splitting_points as fsp 

382 >>> from sympy import sin 

383 >>> from sympy.abc import a, x 

384 >>> fsp(x, x) 

385 {0} 

386 >>> fsp((x-1)**3, x) 

387 {1} 

388 >>> fsp(sin(x+3)*x, x) 

389 {-3, 0}

393 def compute_innermost(expr, res): 

394 if not isinstance(expr, Expr): 

395 return 

396 m = expr.match(p*x + q) 

397 if m and m[p] != 0: 

398 res.add(-m[q]/m[p])

403 compute_innermost(arg, res) 

404 innermost = set() 

405 compute_innermost(expr, innermost) 

406 return innermost

409def _split_mul(f, x): 

410 """ 

411 Split expression ``f`` into fac, po, g, where fac is a constant factor, 

412 po = x**s for some s independent of s, and g is "the rest". 

413 

414 >>> from sympy.integrals.meijerint import _split_mul 

415 >>> from sympy import sin 

416 >>> from sympy.abc import s, x 

417 >>> _split_mul((3*x)**s*sin(x**2)*x, x) 

418 (3**s, x*x**s, sin(x**2)) 

419 """ 

420 from sympy import polarify, unpolarify 

421 fac = S(1) 

422 po = S(1) 

423 g = S(1) 

424 f = expand_power_base(f) 

425 

426 args = Mul.make_args(f) 

427 for a in args: 

428 if a == x: 

429 po *= x 

430 elif x not in a.free_symbols: 

431 fac *= a 

432 else: 

433 if a.is_Pow and x not in a.exp.free_symbols: 

434 c, t = a.base.as_coeff_mul(x) 

435 if t != (x,): 

436 c, t = expand_mul(a.base).as_coeff_mul(x) 

437 if t == (x,): 

438 po *= x**a.exp 

439 fac *= unpolarify(polarify(c**a.exp, subs=False)) 

440 continue 

441 g *= a 

442 

443 return fac, po, g

446def _mul_args(f): 

447 """ 

448 Return a list ``L`` such that Mul(*L) == f. 

449 

450 If f is not a Mul or Pow, L=[f]. 

451 If f=g**n for an integer n, L=[g]*n. 

452 If f is a Mul, L comes from applying _mul_args to all factors of f. 

453 """ 

454 args = Mul.make_args(f) 

455 gs = [] 

456 for g in args: 

457 if g.is_Pow and g.exp.is_Integer: 

458 n = g.exp 

459 base = g.base 

460 if n < 0: 

461 n = -n 

462 base = 1/base 

463 gs += [base]*n 

464 else: 

465 gs.append(g) 

466 return gs

469def _mul_as_two_parts(f): 

470 """ 

471 Find all the ways to split f into a product of two terms. 

472 Return None on failure. 

473 

474 Although the order is canonical from multiset_partitions, this is 

475 not necessarily the best order to process the terms. For example, 

476 if the case of len(gs) == 2 is removed and multiset is allowed to 

477 sort the terms, some tests fail. 

478 

479 >>> from sympy.integrals.meijerint import _mul_as_two_parts 

480 >>> from sympy import sin, exp, ordered 

481 >>> from sympy.abc import x 

482 >>> list(ordered(_mul_as_two_parts(x*sin(x)*exp(x)))) 

483 [(x, exp(x)*sin(x)), (x*exp(x), sin(x)), (x*sin(x), exp(x))] 

484 """ 

485 

486 gs = _mul_args(f) 

487 if len(gs) < 2: 

488 return None 

489 if len(gs) == 2: 

490 return [tuple(gs)] 

491 return [(Mul(*x), Mul(*y)) for (x, y) in multiset_partitions(gs, 2)]

581def _condsimp(cond): 

582 """ 

583 Do naive simplifications on ``cond``. 

584 

585 Note that this routine is completely ad-hoc, simplification rules being 

586 added as need arises rather than following any logical pattern. 

587 

588 >>> from sympy.integrals.meijerint import _condsimp as simp 

589 >>> from sympy import Or, Eq, unbranched_argument as arg, And 

590 >>> from sympy.abc import x, y, z 

591 >>> simp(Or(x < y, z, Eq(x, y))) 

592 z | (x <= y) 

593 >>> simp(Or(x <= y, And(x < y, z))) 

594 x <= y 

595 """ 

596 from sympy import ( 

597 symbols, Wild, Eq, unbranched_argument, exp_polar, pi, I, 

598 arg, periodic_argument, oo, polar_lift) 

599 from sympy.logic.boolalg import BooleanFunction 

600 if not isinstance(cond, BooleanFunction): 

601 return cond 

602 cond = cond.func(*list(map(_condsimp, cond.args))) 

603 change = True 

604 p, q, r = symbols('p q r', cls=Wild) 

605 rules = [ 

606 (Or(p < q, Eq(p, q)), p <= q), 

607 # The next two obviously are instances of a general pattern, but it is 

608 # easier to spell out the few cases we care about. 

609 (And(abs(arg(p)) <= pi, abs(arg(p) - 2*pi) <= pi), 

610 Eq(arg(p) - pi, 0)), 

611 (And(abs(2*arg(p) + pi) <= pi, abs(2*arg(p) - pi) <= pi), 

612 Eq(arg(p), 0)), 

613 (And(abs(unbranched_argument(p)) <= pi, 

614 abs(unbranched_argument(exp_polar(-2*pi*I)*p)) <= pi), 

615 Eq(unbranched_argument(exp_polar(-I*pi)*p), 0)), 

616 (And(abs(unbranched_argument(p)) <= pi/2, 

617 abs(unbranched_argument(exp_polar(-pi*I)*p)) <= pi/2), 

618 Eq(unbranched_argument(exp_polar(-I*pi/2)*p), 0)), 

619 (Or(p <= q, And(p < q, r)), p <= q) 

620 ] 

621 while change: 

622 change = False 

623 for fro, to in rules: 

624 if fro.func != cond.func: 

625 continue 

626 for n, arg1 in enumerate(cond.args): 

627 if r in fro.args[0].free_symbols: 

628 m = arg1.match(fro.args[1]) 

629 num = 1 

630 else: 

631 num = 0 

632 m = arg1.match(fro.args[0]) 

633 if not m: 

634 continue 

635 otherargs = [x.subs(m) for x in fro.args[:num] + fro.args[num + 1:]] 

636 otherlist = [n] 

637 for arg2 in otherargs: 

638 for k, arg3 in enumerate(cond.args): 

639 if k in otherlist: 

640 continue 

641 if arg2 == arg3: 

642 otherlist += [k] 

643 break 

644 if isinstance(arg3, And) and arg2.args[1] == r and \ 

645 isinstance(arg2, And) and arg2.args[0] in arg3.args: 

646 otherlist += [k] 

647 break 

648 if isinstance(arg3, And) and arg2.args[0] == r and \ 

649 isinstance(arg2, And) and arg2.args[1] in arg3.args: 

650 otherlist += [k] 

651 break 

652 if len(otherlist) != len(otherargs) + 1: 

653 continue 

654 newargs = [arg_ for (k, arg_) in enumerate(cond.args) 

655 if k not in otherlist] + [to.subs(m)] 

656 cond = cond.func(*newargs) 

657 change = True 

658 break 

659 # final tweak

1480 if Tuple(*(r1 + (g,))).has(oo, zoo, -oo): 

1481 continue 

1482 g = meijerg(g.an, g.aother, g.bm, g.bother, 

1483 unpolarify(g.argument, exponents_only=True)) 

1484 res.append(r1 + (g,)) 

1485 if res: 

1486 return res, cond 

1487 

1488 # try recursive mellin transform 

1489 if not recursive: 

1490 return None 

1491 _debug('Trying recursive Mellin transform method.') 

1492 from sympy.integrals.transforms import (mellin_transform, 

1493 inverse_mellin_transform, IntegralTransformError, 

1494 MellinTransformStripError) 

1495 from sympy import oo, nan, zoo, simplify, cancel

1497 def my_imt(F, s, x, strip): 

1498 """ Calling simplify() all the time is slow and not helpful, since 

1499 most of the time it only factors things in a way that has to be 

1500 un-done anyway. But sometimes it can remove apparent poles. """ 

1501 # XXX should this be in inverse_mellin_transform? 

1502 try: 

1503 return inverse_mellin_transform(F, s, x, strip, 

1504 as_meijerg=True, needeval=True) 

1505 except MellinTransformStripError: 

1506 return inverse_mellin_transform( 

1507 simplify(cancel(expand(F))), s, x, strip, 

1508 as_meijerg=True, needeval=True) 

1509 f = f_ 

1510 s = _dummy('s', 'rewrite-single', f) 

1511 # to avoid infinite recursion, we have to force the two g functions case 

1512 

1513 def my_integrator(f, x): 

1514 from sympy import Integral, hyperexpand 

1515 r = _meijerint_definite_4(f, x, only_double=True) 

1516 if r is not None: 

1517 res, cond = r 

1518 res = _my_unpolarify(hyperexpand(res, rewrite='nonrepsmall')) 

1519 return Piecewise((res, cond), 

1520 (Integral(f, (x, 0, oo)), True)) 

1521 return Integral(f, (x, 0, oo)) 

1522 try: 

1523 F, strip, _ = mellin_transform(f, x, s, integrator=my_integrator, 

1524 simplify=False, needeval=True) 

1525 g = my_imt(F, s, x, strip) 

1526 except IntegralTransformError: 

1527 g = None 

1528 if g is None: 

1529 # We try to find an expression by analytic continuation. 

1530 # (also if the dummy is already in the expression, there is no point in 

1531 # putting in another one) 

1532 a = _dummy_('a', 'rewrite-single') 

1533 if a not in f.free_symbols and _is_analytic(f, x): 

1534 try: 

1535 F, strip, _ = mellin_transform(f.subs(x, a*x), x, s, 

1536 integrator=my_integrator, 

1537 needeval=True, simplify=False) 

1538 g = my_imt(F, s, x, strip).subs(a, 1) 

1539 except IntegralTransformError: 

1540 g = None 

1541 if g is None or g.has(oo, nan, zoo): 

1542 _debug('Recursive Mellin transform failed.') 

1543 return None 

1544 args = Add.make_args(g) 

1545 res = [] 

1546 for f in args: 

1547 c, m = f.as_coeff_mul(x) 

1548 if len(m) > 1: 

1549 raise NotImplementedError('Unexpected form...') 

1550 g = m[0] 

1551 a, b = _get_coeff_exp(g.argument, x) 

1552 res += [(c, 0, meijerg(g.an, g.aother, g.bm, g.bother, 

1553 unpolarify(polarify( 

1554 a, lift=True), exponents_only=True) 

1555 *x**b))] 

1556 _debug('Recursive Mellin transform worked:', g) 

1557 return res, True

1574def _rewrite2(f, x): 

1575 """ 

1576 Try to rewrite f as a product of two G functions of arguments a*x**b. 

1577 Return fac, po, g1, g2 such that f = fac*po*g1*g2, where fac is 

1578 independent of x and po is x**s. 

1579 Here g1 and g2 are results of _rewrite_single. 

1580 Returns None on failure. 

1581 """ 

1582 fac, po, g = _split_mul(f, x) 

1583 if any(_rewrite_single(expr, x, False) is None for expr in _mul_args(g)): 

1584 return None 

1585 l = _mul_as_two_parts(g) 

1586 if not l: 

1587 return None 

1588 l = list(ordered(l, [ 

1589 lambda p: max(len(_exponents(p[0], x)), len(_exponents(p[1], x))), 

1590 lambda p: max(len(_functions(p[0], x)), len(_functions(p[1], x))), 

1591 lambda p: max(len(_find_splitting_points(p[0], x)), 

1592 len(_find_splitting_points(p[1], x)))])) 

1593 

1594 for recursive in [False, True]: 

1595 for fac1, fac2 in l: 

1596 g1 = _rewrite_single(fac1, x, recursive) 

1597 g2 = _rewrite_single(fac2, x, recursive) 

1598 if g1 and g2: 

1599 cond = And(g1[1], g2[1]) 

1600 if cond != False: 

1601 return fac, po, g1[0], g2[0], cond

1641def _meijerint_indefinite_1(f, x): 

1642 """ Helper that does not attempt any substitution. """ 

1643 from sympy import Integral, piecewise_fold, nan, zoo 

1644 _debug('Trying to compute the indefinite integral of', f, 'wrt', x) 

1645 

1646 gs = _rewrite1(f, x) 

1647 if gs is None: 

1648 # Note: the code that calls us will do expand() and try again 

1649 return None 

1650 

1651 fac, po, gl, cond = gs 

1652 _debug(' could rewrite:', gs) 

1653 res = S(0) 

1654 for C, s, g in gl: 

1655 a, b = _get_coeff_exp(g.argument, x) 

1656 _, c = _get_coeff_exp(po, x) 

1657 c += s 

1658 

1659 # we do a substitution t=a*x**b, get integrand fac*t**rho*g 

1660 fac_ = fac * C / (b*a**((1 + c)/b)) 

1661 rho = (c + 1)/b - 1 

1662 

1663 # we now use t**rho*G(params, t) = G(params + rho, t) 

1664 # [L, page 150, equation (4)] 

1665 # and integral G(params, t) dt = G(1, params+1, 0, t) 

1666 # (or a similar expression with 1 and 0 exchanged ... pick the one 

1667 # which yields a well-defined function) 

1668 # [R, section 5] 

1669 # (Note that this dummy will immediately go away again, so we 

1670 # can safely pass S(1) for ``expr``.) 

1671 t = _dummy('t', 'meijerint-indefinite', S(1)) 

1672 

1673 def tr(p): 

1674 return [a + rho + 1 for a in p] 

1675 if any(b.is_integer and (b <= 0) == True for b in tr(g.bm)): 

1676 r = -meijerg( 

1677 tr(g.an), tr(g.aother) + [1], tr(g.bm) + [0], tr(g.bother), t) 

1678 else: 

1679 r = meijerg( 

1680 tr(g.an) + [1], tr(g.aother), tr(g.bm), tr(g.bother) + [0], t) 

1681 # The antiderivative is most often expected to be defined 

1682 # in the neighborhood of x = 0. 

1683 place = 0 

1684 if b < 0 or f.subs(x, 0).has(nan, zoo): 

1685 place = None 

1686 r = hyperexpand(r.subs(t, a*x**b), place=place) 

1687 

1688 # now substitute back 

1689 # Note: we really do want the powers of x to combine. 

1690 res += powdenest(fac_*r, polar=True)

1692 def _clean(res): 

1693 """This multiplies out superfluous powers of x we created, and chops off 

1694 constants: 

1695 

1696 >> _clean(x*(exp(x)/x - 1/x) + 3) 

1697 exp(x) 

1698 

1699 cancel is used before mul_expand since it is possible for an 

1700 expression to have an additive constant that doesn't become isolated 

1701 with simple expansion. Such a situation was identified in issue 6369: 

1702 

1703 

1704 >>> from sympy import sqrt, cancel 

1705 >>> from sympy.abc import x 

1706 >>> a = sqrt(2*x + 1) 

1707 >>> bad = (3*x*a**5 + 2*x - a**5 + 1)/a**2 

1708 >>> bad.expand().as_independent(x)[0] 

1709 0 

1710 >>> cancel(bad).expand().as_independent(x)[0] 

1711 1 

1712 """ 

1713 from sympy import cancel 

1714 res = expand_mul(cancel(res), deep=False) 

1715 return Add._from_args(res.as_coeff_add(x)[1]) 

1716 

1717 res = piecewise_fold(res) 

1718 if res.is_Piecewise: 

1719 newargs = [] 

1720 for expr, cond in res.args: 

1721 expr = _my_unpolarify(_clean(expr)) 

1722 newargs += [(expr, cond)] 

1723 res = Piecewise(*newargs) 

1724 else: 

1725 res = _my_unpolarify(_clean(res)) 

1726 return Piecewise((res, _my_unpolarify(cond)), (Integral(f, x), True))

1729@timeit 

1730def meijerint_definite(f, x, a, b): 

1731 """ 

1732 Integrate ``f`` over the interval [``a``, ``b``], by rewriting it as a product 

1733 of two G functions, or as a single G function. 

1734 

1735 Return res, cond, where cond are convergence conditions. 

1736 

1737 Examples 

1738 ======== 

1739 

1740 >>> from sympy.integrals.meijerint import meijerint_definite 

1741 >>> from sympy import exp, oo 

1742 >>> from sympy.abc import x 

1743 >>> meijerint_definite(exp(-x**2), x, -oo, oo) 

1744 (sqrt(pi), True) 

1745 

1746 This function is implemented as a succession of functions 

1747 meijerint_definite, _meijerint_definite_2, _meijerint_definite_3, 

1748 _meijerint_definite_4. Each function in the list calls the next one 

1749 (presumably) several times. This means that calling meijerint_definite 

1750 can be very costly. 

1751 """ 

1752 # This consists of three steps: 

1753 # 1) Change the integration limits to 0, oo 

1754 # 2) Rewrite in terms of G functions 

1755 # 3) Evaluate the integral 

1756 # 

1757 # There are usually several ways of doing this, and we want to try all. 

1758 # This function does (1), calls _meijerint_definite_2 for step (2). 

1759 from sympy import arg, exp, I, And, DiracDelta, SingularityFunction 

1760 _debug('Integrating', f, 'wrt %s from %s to %s.' % (x, a, b)) 

1761 

1762 if f.has(DiracDelta): 

1763 _debug('Integrand has DiracDelta terms - giving up.') 

1764 return None 

1765 

1766 if f.has(SingularityFunction): 

1767 _debug('Integrand has Singularity Function terms - giving up.') 

1768 return None 

1769 

1770 f_, x_, a_, b_ = f, x, a, b 

1771 

1772 # Let's use a dummy in case any of the boundaries has x. 

1773 d = Dummy('x') 

1774 f = f.subs(x, d) 

1775 x = d 

1776 

1777 if a == b: 

1778 return (S.Zero, True) 

1779 

1780 results = [] 

1781 if a == -oo and b != oo: 

1782 return meijerint_definite(f.subs(x, -x), x, -b, -a) 

1783 

1784 elif a == -oo: 

1785 # Integrating -oo to oo. We need to find a place to split the integral. 

1786 _debug(' Integrating -oo to +oo.') 

1787 innermost = _find_splitting_points(f, x) 

1788 _debug(' Sensible splitting points:', innermost) 

1789 for c in sorted(innermost, key=default_sort_key, reverse=True) + [S(0)]: 

1790 _debug(' Trying to split at', c) 

1791 if not c.is_real: 

1792 _debug(' Non-real splitting point.') 

1793 continue 

1794 res1 = _meijerint_definite_2(f.subs(x, x + c), x) 

1795 if res1 is None: 

1796 _debug(' But could not compute first integral.') 

1797 continue 

1798 res2 = _meijerint_definite_2(f.subs(x, c - x), x) 

1799 if res2 is None: 

1800 _debug(' But could not compute second integral.') 

1801 continue 

1802 res1, cond1 = res1 

1803 res2, cond2 = res2 

1804 cond = _condsimp(And(cond1, cond2)) 

1805 if cond == False: 

1806 _debug(' But combined condition is always false.') 

1807 continue 

1808 res = res1 + res2 

1809 return res, cond 

1810 

1811 elif a == oo: 

1812 res = meijerint_definite(f, x, b, oo) 

1813 return -res[0], res[1] 

1814 

1815 elif (a, b) == (0, oo): 

1816 # This is a common case - try it directly first. 

1817 res = _meijerint_definite_2(f, x) 

1818 if res: 

1819 if _has(res[0], meijerg): 

1820 results.append(res) 

1821 else: 

1822 return res 

1823 

1824 else: 

1825 if b == oo: 

1826 for split in _find_splitting_points(f, x): 

1827 if (a - split >= 0) == True: 

1828 _debug('Trying x -> x + %s' % split) 

1829 res = _meijerint_definite_2(f.subs(x, x + split) 

1830 *Heaviside(x + split - a), x) 

1831 if res: 

1832 if _has(res[0], meijerg): 

1833 results.append(res) 

1834 else: 

1835 return res 

1836 

1837 f = f.subs(x, x + a) 

1838 b = b - a 

1839 a = 0 

1840 if b != oo: 

1841 phi = exp(I*arg(b)) 

1842 b = abs(b) 

1843 f = f.subs(x, phi*x) 

1844 f *= Heaviside(b - x)*phi 

1845 b = oo 

1846 

1847 _debug('Changed limits to', a, b) 

1848 _debug('Changed function to', f) 

1849 res = _meijerint_definite_2(f, x) 

1850 if res: 

1851 if _has(res[0], meijerg): 

1852 results.append(res) 

1853 else: 

1854 return res 

1855 if f_.has(HyperbolicFunction): 

1856 _debug('Try rewriting hyperbolics in terms of exp.') 

1857 rv = meijerint_definite( 

1858 _rewrite_hyperbolics_as_exp(f_), x_, a_, b_) 

1859 if rv: 

1860 if not type(rv) is list: 

1861 rv = (collect(factor_terms(rv[0]), rv[0].atoms(exp)),) + rv[1:] 

1862 return rv 

1863 results.extend(rv) 

1864 if results: 

1865 return next(ordered(results))

1868def _guess_expansion(f, x): 

1869 """ Try to guess sensible rewritings for integrand f(x). """ 

1870 from sympy import expand_trig 

1871 from sympy.functions.elementary.trigonometric import TrigonometricFunction 

1872 res = [(f, 'original integrand')] 

1873 

1874 orig = res[-1][0] 

1875 saw = {orig} 

1876 expanded = expand_mul(orig) 

1877 if expanded not in saw: 

1878 res += [(expanded, 'expand_mul')] 

1879 saw.add(expanded) 

1880 

1881 expanded = expand(orig) 

1882 if expanded not in saw: 

1883 res += [(expanded, 'expand')] 

1884 saw.add(expanded) 

1885 

1886 if orig.has(TrigonometricFunction, HyperbolicFunction): 

1887 expanded = expand_mul(expand_trig(orig)) 

1888 if expanded not in saw: 

1889 res += [(expanded, 'expand_trig, expand_mul')] 

1890 saw.add(expanded) 

1891 

1892 if orig.has(cos, sin): 

1893 reduced = sincos_to_sum(orig) 

1894 if reduced not in saw: 

1895 res += [(reduced, 'trig power reduction')] 

1896 saw.add(reduced) 

1897 

1898 return res

1903 Try to integrate f dx from zero to infinty. 

1904 

1905 The body of this function computes various 'simplifications' 

1906 f1, f2, ... of f (e.g. by calling expand_mul(), trigexpand() 

1907 - see _guess_expansion) and calls _meijerint_definite_3 with each of 

1908 these in succession. 

1909 If _meijerint_definite_3 succeeds with any of the simplified functions, 

1910 returns this result. 

1911 """

1913 # _meijerint_definite_3 for (2) and (3) combined. 

1914

1916 # XXX if a nonnegative symbol is used there will be test failures

1918 f = f.subs(x, dummy) 

1919 x = dummy 

1920 

1921 if f == 0: 

1922 return S(0), True 

1923 

1924 for g, explanation in _guess_expansion(f, x): 

1925 _debug('Trying', explanation) 

1926 res = _meijerint_definite_3(g, x) 

1927 if res: 

1928 return res

1960@timeit 

1961def _meijerint_definite_4(f, x, only_double=False): 

1962 """ 

1963 Try to integrate f dx from zero to infinity. 

1964 

1965 This function tries to apply the integration theorems found in literature, 

1966 i.e. it tries to rewrite f as either one or a product of two G-functions. 

1967 

1968 The parameter ``only_double`` is used internally in the recursive algorithm 

1969 to disable trying to rewrite f as a single G-function. 

1970 """ 

1971 # This function does (2) and (3) 

1972 _debug('Integrating', f) 

1973 # Try single G function. 

1974 if not only_double: 

1975 gs = _rewrite1(f, x, recursive=False) 

1976 if gs is not None: 

1977 fac, po, g, cond = gs 

1978 _debug('Could rewrite as single G function:', fac, po, g) 

1979 res = S(0) 

1980 for C, s, f in g: 

1981 if C == 0: 

1982 continue 

1983 C, f = _rewrite_saxena_1(fac*C, po*x**s, f, x) 

1984 res += C*_int0oo_1(f, x) 

1985 cond = And(cond, _check_antecedents_1(f, x)) 

1986 if cond == False: 

1987 break 

1988 cond = _my_unpolarify(cond) 

1989 if cond == False: 

1990 _debug('But cond is always False.') 

1991 else: 

1992 _debug('Result before branch substitutions is:', res) 

1993 return _my_unpolarify(hyperexpand(res)), cond 

1994 

1995 # Try two G functions. 

1996 gs = _rewrite2(f, x) 

1997 if gs is not None: 

1998 for full_pb in [False, True]: 

1999 fac, po, g1, g2, cond = gs 

2000 _debug('Could rewrite as two G functions:', fac, po, g1, g2) 

2001 res = S(0) 

2002 for C1, s1, f1 in g1: 

2003 for C2, s2, f2 in g2: 

2004 r = _rewrite_saxena(fac*C1*C2, po*x**(s1 + s2), 

2005 f1, f2, x, full_pb) 

2006 if r is None: 

2007 _debug('Non-rational exponents.') 

2008 return 

2009 C, f1_, f2_ = r 

2010 _debug('Saxena subst for yielded:', C, f1_, f2_) 

2011 cond = And(cond, _check_antecedents(f1_, f2_, x)) 

2012 if cond == False: 

2013 break 

2014 res += C*_int0oo(f1_, f2_, x) 

2015 else: 

2016 continue 

2017 break 

2018 cond = _my_unpolarify(cond) 

2019 if cond == False: 

2020 _debug('But cond is always False (full_pb=%s).' % full_pb) 

2021 else: 

2022 _debug('Result before branch substitutions is:', res) 

2023 if only_double: 

2024 return res, cond 

2025 return _my_unpolarify(hyperexpand(res)), cond

2028def meijerint_inversion(f, x, t): 

2029 r""" 

2030 Compute the inverse laplace transform 

2031 :math:\int_{c+i\infty}^{c-i\infty} f(x) e^{tx) dx, 

2032 for real c larger than the real part of all singularities of f. 

2033 Note that ``t`` is always assumed real and positive. 

2034 

2035 Return None if the integral does not exist or could not be evaluated. 

2036 

2037 Examples 

2038 ======== 

2039 

2040 >>> from sympy.abc import x, t 

2041 >>> from sympy.integrals.meijerint import meijerint_inversion 

2042 >>> meijerint_inversion(1/x, x, t) 

2043 Heaviside(t) 

2044 """ 

2045 from sympy import I, Integral, exp, expand, log, Add, Mul, Heaviside 

2046 f_ = f 

2047 t_ = t 

2048 t = Dummy('t', polar=True) # We don't want sqrt(t**2) = abs(t) etc 

2049 f = f.subs(t_, t) 

2050 _debug('Laplace-inverting', f) 

2051 if not _is_analytic(f, x): 

2052 _debug('But expression is not analytic.') 

2053 return None 

2054 # Exponentials correspond to shifts; we filter them out and then 

2055 # shift the result later. If we are given an Add this will not 

2056 # work, but the calling code will take care of that.
