200 def _print_sign(self, expr): 

201 from sympy import Abs 

202 arg, = expr.args 

203 if arg.is_integer: 

204 new_expr = merge(0, isign(1, arg), Eq(arg, 0)) 

205 elif arg.is_complex: 

206 new_expr = merge(cmplx(literal_dp(0), literal_dp(0)), arg/Abs(arg), Eq(Abs(arg), literal_dp(0)))

208 new_expr = merge(literal_dp(0), dsign(literal_dp(1), arg), Eq(arg, literal_dp(0)))

212 def _print_Piecewise(self, expr): 

213 if expr.args[-1].cond != True: 

214 # We need the last conditional to be a True, otherwise the resulting 

215 # function may not return a result. 

216 raise ValueError("All Piecewise expressions must contain an " 

217 "(expr, True) statement to be used as a default " 

218 "condition. Without one, the generated " 

219 "expression may not evaluate to anything under " 

220 "some condition.") 

221 lines = [] 

222 if expr.has(Assignment): 

223 for i, (e, c) in enumerate(expr.args): 

224 if i == 0: 

225 lines.append("if (%s) then" % self._print(c)) 

226 elif i == len(expr.args) - 1 and c == True: 

227 lines.append("else") 

228 else: 

229 lines.append("else if (%s) then" % self._print(c)) 

230 lines.append(self._print(e)) 

231 lines.append("end if") 

232 return "\n".join(lines) 

233 elif self._settings["standard"] >= 95: 

234 # Only supported in F95 and newer: 

235 # The piecewise was used in an expression, need to do inline 

236 # operators. This has the downside that inline operators will 

237 # not work for statements that span multiple lines (Matrix or 

238 # Indexed expressions). 

239 pattern = "merge({T}, {F}, {COND})" 

240 code = self._print(expr.args[-1].expr) 

241 terms = list(expr.args[:-1]) 

242 while terms: 

243 e, c = terms.pop() 

244 expr = self._print(e) 

245 cond = self._print(c) 

246 code = pattern.format(T=expr, F=code, COND=cond) 

247 return code 

248 else: 

249 # `merge` is not supported prior to F95 

250 raise NotImplementedError("Using Piecewise as an expression using " 

251 "inline operators is not supported in " 

252 "standards earlier than Fortran95.")

258 def _print_Add(self, expr): 

259 # purpose: print complex numbers nicely in Fortran. 

260 # collect the purely real and purely imaginary parts: 

261 pure_real = [] 

262 pure_imaginary = [] 

263 mixed = [] 

264 for arg in expr.args: 

265 if arg.is_number and arg.is_real: 

266 pure_real.append(arg) 

267 elif arg.is_number and arg.is_imaginary: 

268 pure_imaginary.append(arg) 

269 else: 

270 mixed.append(arg) 

271 if pure_imaginary: 

272 if mixed: 

273 PREC = precedence(expr) 

274 term = Add(*mixed) 

275 t = self._print(term) 

276 if t.startswith('-'): 

277 sign = "-" 

278 t = t[1:] 

279 else: 

280 sign = "+" 

281 if precedence(term) < PREC: 

282 t = "(%s)" % t 

283 

284 return "cmplx(%s,%s) %s %s" % ( 

285 self._print(Add(*pure_real)), 

286 self._print(-S.ImaginaryUnit*Add(*pure_imaginary)), 

287 sign, t, 

288 ) 

289 else: 

290 return "cmplx(%s,%s)" % ( 

291 self._print(Add(*pure_real)), 

292 self._print(-S.ImaginaryUnit*Add(*pure_imaginary)), 

293 ) 

294 else: 

295 return CodePrinter._print_Add(self, expr)

507 def _wrap_fortran(self, lines): 

508 """Wrap long Fortran lines 

509 

510 Argument: 

511 lines -- a list of lines (without \\n character) 

512 

513 A comment line is split at white space. Code lines are split with a more 

514 complex rule to give nice results. 

515 """ 

516 # routine to find split point in a code line 

517 my_alnum = set("_+-." + string.digits + string.ascii_letters) 

518 my_white = set(" \t()") 

519 

520 def split_pos_code(line, endpos): 

521 if len(line) <= endpos: 

522 return len(line) 

523 pos = endpos 

524 split = lambda pos: \ 

525 (line[pos] in my_alnum and line[pos - 1] not in my_alnum) or \ 

526 (line[pos] not in my_alnum and line[pos - 1] in my_alnum) or \ 

527 (line[pos] in my_white and line[pos - 1] not in my_white) or \ 

528 (line[pos] not in my_white and line[pos - 1] in my_white) 

529 while not split(pos): 

530 pos -= 1 

531 if pos == 0: 

532 return endpos 

533 return pos 

534 # split line by line and add the split lines to result 

535 result = [] 

536 if self._settings['source_format'] == 'free': 

537 trailing = ' &' 

538 else: 

539 trailing = '' 

540 for line in lines: 

541 if line.startswith(self._lead['comment']): 

542 # comment line 

543 if len(line) > 72: 

544 pos = line.rfind(" ", 6, 72) 

545 if pos == -1: 

546 pos = 72 

547 hunk = line[:pos] 

548 line = line[pos:].lstrip() 

549 result.append(hunk) 

550 while line: 

551 pos = line.rfind(" ", 0, 66) 

552 if pos == -1 or len(line) < 66: 

553 pos = 66 

554 hunk = line[:pos] 

555 line = line[pos:].lstrip() 

556 result.append("%s%s" % (self._lead['comment'], hunk)) 

557 else: 

558 result.append(line) 

559 elif line.startswith(self._lead['code']): 

560 # code line 

561 pos = split_pos_code(line, 72) 

562 hunk = line[:pos].rstrip() 

563 line = line[pos:].lstrip() 

564 if line: 

565 hunk += trailing 

566 result.append(hunk) 

567 while line: 

568 pos = split_pos_code(line, 65) 

569 hunk = line[:pos].rstrip() 

570 line = line[pos:].lstrip() 

571 if line: 

572 hunk += trailing 

573 result.append("%s%s" % (self._lead['cont'], hunk)) 

574 else: 

575 result.append(line) 

576 return result

578 def indent_code(self, code): 

579 """Accepts a string of code or a list of code lines""" 

580 if isinstance(code, string_types): 

581 code_lines = self.indent_code(code.splitlines(True)) 

582 return ''.join(code_lines) 

583 

584 free = self._settings['source_format'] == 'free' 

585 code = [ line.lstrip(' \t') for line in code ] 

586 

587 inc_keyword = ('do ', 'if(', 'if ', 'do\n', 'else', 'program', 'interface') 

588 dec_keyword = ('end do', 'enddo', 'end if', 'endif', 'else', 'end program', 'end interface') 

589 

590 increase = [ int(any(map(line.startswith, inc_keyword))) 

591 for line in code ] 

592 decrease = [ int(any(map(line.startswith, dec_keyword))) 

593 for line in code ] 

594 continuation = [ int(any(map(line.endswith, ['&', '&\n']))) 

595 for line in code ] 

596 

597 level = 0 

598 cont_padding = 0 

599 tabwidth = 3 

600 new_code = [] 

601 for i, line in enumerate(code): 

602 if line == '' or line == '\n': 

603 new_code.append(line) 

604 continue 

605 level -= decrease[i] 

606 

607 if free: 

608 padding = " "*(level*tabwidth + cont_padding) 

609 else: 

610 padding = " "*level*tabwidth 

611 

612 line = "%s%s" % (padding, line) 

613 if not free: 

614 line = self._pad_leading_columns([line])[0] 

615 

616 new_code.append(line) 

617 

618 if continuation[i]: 

619 cont_padding = 2*tabwidth 

620 else: 

621 cont_padding = 0 

622 level += increase[i] 

623 

624 if not free: 

625 return self._wrap_fortran(new_code) 

626 return new_code

775def fcode(expr, assign_to=None, **settings): 

776 """Converts an expr to a string of fortran code 

777 

778 Parameters 

779 ========== 

780 

781 expr : Expr 

782 A sympy expression to be converted. 

783 assign_to : optional 

784 When given, the argument is used as the name of the variable to which 

785 the expression is assigned. Can be a string, ``Symbol``, 

786 ``MatrixSymbol``, or ``Indexed`` type. This is helpful in case of 

787 line-wrapping, or for expressions that generate multi-line statements. 

788 precision : integer, optional 

789 DEPRECATED. Use type_mappings instead. The precision for numbers such 

790 as pi [default=17]. 

791 user_functions : dict, optional 

792 A dictionary where keys are ``FunctionClass`` instances and values are 

793 their string representations. Alternatively, the dictionary value can 

794 be a list of tuples i.e. [(argument_test, cfunction_string)]. See below 

795 for examples. 

796 human : bool, optional 

797 If True, the result is a single string that may contain some constant 

798 declarations for the number symbols. If False, the same information is 

799 returned in a tuple of (symbols_to_declare, not_supported_functions, 

800 code_text). [default=True]. 

801 contract: bool, optional 

802 If True, ``Indexed`` instances are assumed to obey tensor contraction 

803 rules and the corresponding nested loops over indices are generated. 

804 Setting contract=False will not generate loops, instead the user is 

805 responsible to provide values for the indices in the code. 

806 [default=True]. 

807 source_format : optional 

808 The source format can be either 'fixed' or 'free'. [default='fixed'] 

809 standard : integer, optional 

810 The Fortran standard to be followed. This is specified as an integer. 

811 Acceptable standards are 66, 77, 90, 95, 2003, and 2008. Default is 77. 

812 Note that currently the only distinction internally is between 

813 standards before 95, and those 95 and after. This may change later as 

814 more features are added. 

815 name_mangling : bool, optional 

816 If True, then the variables that would become identical in 

817 case-insensitive Fortran are mangled by appending different number 

818 of ``_`` at the end. If False, SymPy won't interfere with naming of 

819 variables. [default=True] 

820 

821 Examples 

822 ======== 

823 

824 >>> from sympy import fcode, symbols, Rational, sin, ceiling, floor 

825 >>> x, tau = symbols("x, tau") 

826 >>> fcode((2*tau)**Rational(7, 2)) 

827 ' 8*sqrt(2.0d0)*tau**(7.0d0/2.0d0)' 

828 >>> fcode(sin(x), assign_to="s") 

829 ' s = sin(x)' 

830 

831 Custom printing can be defined for certain types by passing a dictionary of 

832 "type" : "function" to the ``user_functions`` kwarg. Alternatively, the 

833 dictionary value can be a list of tuples i.e. [(argument_test, 

834 cfunction_string)]. 

835 

836 >>> custom_functions = { 

837 ... "ceiling": "CEIL", 

838 ... "floor": [(lambda x: not x.is_integer, "FLOOR1"), 

839 ... (lambda x: x.is_integer, "FLOOR2")] 

840 ... } 

841 >>> fcode(floor(x) + ceiling(x), user_functions=custom_functions) 

842 ' CEIL(x) + FLOOR1(x)' 

843 

844 ``Piecewise`` expressions are converted into conditionals. If an 

845 ``assign_to`` variable is provided an if statement is created, otherwise 

846 the ternary operator is used. Note that if the ``Piecewise`` lacks a 

847 default term, represented by ``(expr, True)`` then an error will be thrown. 

848 This is to prevent generating an expression that may not evaluate to 

849 anything. 

850 

851 >>> from sympy import Piecewise 

852 >>> expr = Piecewise((x + 1, x > 0), (x, True)) 

853 >>> print(fcode(expr, tau)) 

854 if (x > 0) then 

855 tau = x + 1 

856 else 

857 tau = x 

858 end if 

859 

860 Support for loops is provided through ``Indexed`` types. With 

861 ``contract=True`` these expressions will be turned into loops, whereas 

862 ``contract=False`` will just print the assignment expression that should be 

863 looped over: 

864 

865 >>> from sympy import Eq, IndexedBase, Idx 

866 >>> len_y = 5 

867 >>> y = IndexedBase('y', shape=(len_y,)) 

868 >>> t = IndexedBase('t', shape=(len_y,)) 

869 >>> Dy = IndexedBase('Dy', shape=(len_y-1,)) 

870 >>> i = Idx('i', len_y-1) 

871 >>> e=Eq(Dy[i], (y[i+1]-y[i])/(t[i+1]-t[i])) 

872 >>> fcode(e.rhs, assign_to=e.lhs, contract=False) 

873 ' Dy(i) = (y(i + 1) - y(i))/(t(i + 1) - t(i))' 

874 

875 Matrices are also supported, but a ``MatrixSymbol`` of the same dimensions 

876 must be provided to ``assign_to``. Note that any expression that can be 

877 generated normally can also exist inside a Matrix: 

878 

879 >>> from sympy import Matrix, MatrixSymbol 

880 >>> mat = Matrix([x**2, Piecewise((x + 1, x > 0), (x, True)), sin(x)]) 

881 >>> A = MatrixSymbol('A', 3, 1) 

882 >>> print(fcode(mat, A)) 

883 A(1, 1) = x**2 

884 if (x > 0) then 

885 A(2, 1) = x + 1 

886 else 

887 A(2, 1) = x 

888 end if 

889 A(3, 1) = sin(x) 

890 """ 

891 

892 return FCodePrinter(settings).doprint(expr, assign_to)
