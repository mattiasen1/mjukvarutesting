187 @property 

188 def canonical(self): 

189 """Return a canonical form of the relational by putting a 

190 Number on the rhs else ordering the args. The relation is also changed 

191 so that the left-hand side expression does not start with a `-`. 

192 No other simplification is attempted. 

193 

194 Examples 

195 ======== 

196 

197 >>> from sympy.abc import x, y 

198 >>> x < 2 

199 x < 2 

200 >>> _.reversed.canonical 

201 x < 2 

202 >>> (-y < x).canonical 

203 x > -y 

204 >>> (-y > x).canonical 

205 x < -y 

206 """ 

207 args = self.args 

208 r = self 

209 if r.rhs.is_number: 

210 if r.rhs.is_Number and r.lhs.is_Number and r.lhs > r.rhs: 

211 r = r.reversed 

212 elif r.lhs.is_number: 

213 r = r.reversed 

214 elif tuple(ordered(args)) != args: 

215 r = r.reversed 

216 

217 # Check if first value has negative sign 

218 if not isinstance(r.lhs, BooleanAtom) and \ 

219 r.lhs.could_extract_minus_sign(): 

220 r = r.reversedsign 

221 elif not isinstance(r.rhs, BooleanAtom) and not r.rhs.is_number and \ 

222 r.rhs.could_extract_minus_sign(): 

223 # Right hand side has a minus, but not lhs. 

224 # How does the expression with reversed signs behave? 

225 # This is so that expressions of the type Eq(x, -y) and Eq(-x, y) 

226 # have the same canonical representation 

227 expr1, _ = ordered([r.lhs, -r.rhs]) 

228 if expr1 != r.lhs: 

229 r = r.reversed.reversedsign 

230 return r

232 def equals(self, other, failing_expression=False): 

233 """Return True if the sides of the relationship are mathematically 

234 identical and the type of relationship is the same. 

235 If failing_expression is True, return the expression whose truth value 

236 was unknown.""" 

237 if isinstance(other, Relational): 

238 if self == other or self.reversed == other: 

239 return True 

240 a, b = self, other 

241 if a.func in (Eq, Ne) or b.func in (Eq, Ne): 

242 if a.func != b.func: 

243 return False 

244 left, right = [i.equals(j, 

245 failing_expression=failing_expression) 

246 for i, j in zip(a.args, b.args)] 

247 if left is True: 

248 return right 

249 if right is True: 

250 return left 

251 lr, rl = [i.equals(j, failing_expression=failing_expression) 

252 for i, j in zip(a.args, b.reversed.args)] 

253 if lr is True: 

254 return rl 

255 if rl is True: 

256 return lr 

257 e = (left, right, lr, rl) 

258 if all(i is False for i in e): 

259 return False 

260 for i in e: 

261 if i not in (True, False): 

262 return i 

263 else: 

264 if b.func != a.func: 

265 b = b.reversed 

266 if a.func != b.func: 

267 return False 

268 left = a.lhs.equals(b.lhs, 

269 failing_expression=failing_expression) 

270 if left is False: 

271 return False 

272 right = a.rhs.equals(b.rhs, 

273 failing_expression=failing_expression) 

274 if right is False: 

275 return False 

276 if left is True: 

277 return right 

278 return left

325class Equality(Relational): 

326 """An equal relation between two objects. 

327 

328 Represents that two objects are equal. If they can be easily shown 

329 to be definitively equal (or unequal), this will reduce to True (or 

330 False). Otherwise, the relation is maintained as an unevaluated 

331 Equality object. Use the ``simplify`` function on this object for 

332 more nontrivial evaluation of the equality relation. 

333 

334 As usual, the keyword argument ``evaluate=False`` can be used to 

335 prevent any evaluation. 

336 

337 Examples 

338 ======== 

339 

340 >>> from sympy import Eq, simplify, exp, cos 

341 >>> from sympy.abc import x, y 

342 >>> Eq(y, x + x**2) 

343 Eq(y, x**2 + x) 

344 >>> Eq(2, 5) 

345 False 

346 >>> Eq(2, 5, evaluate=False) 

347 Eq(2, 5) 

348 >>> _.doit() 

349 False 

350 >>> Eq(exp(x), exp(x).rewrite(cos)) 

351 Eq(exp(x), sinh(x) + cosh(x)) 

352 >>> simplify(_) 

353 True 

354 

355 See Also 

356 ======== 

357 

358 sympy.logic.boolalg.Equivalent : for representing equality between two 

359 boolean expressions 

360 

361 Notes 

362 ===== 

363 

364 This class is not the same as the == operator. The == operator tests 

365 for exact structural equality between two expressions; this class 

366 compares expressions mathematically. 

367 

368 If either object defines an `_eval_Eq` method, it can be used in place of 

369 the default algorithm. If `lhs._eval_Eq(rhs)` or `rhs._eval_Eq(lhs)` 

370 returns anything other than None, that return value will be substituted for 

371 the Equality. If None is returned by `_eval_Eq`, an Equality object will 

372 be created as usual. 

373 

374 Since this object is already an expression, it does not respond to 

375 the method `as_expr` if one tries to create `x - y` from Eq(x, y). 

376 This can be done with the `rewrite(Add)` method. 

377 """ 

378 rel_op = '=='

393 evaluate = options.pop('evaluate', global_evaluate[0])

395 if evaluate: 

396 # If one expression has an _eval_Eq, return its results. 

397 if hasattr(lhs, '_eval_Eq'): 

398 r = lhs._eval_Eq(rhs) 

399 if r is not None: 

400 return r 

401 if hasattr(rhs, '_eval_Eq'): 

402 r = rhs._eval_Eq(lhs) 

403 if r is not None: 

404 return r 

405 # If expressions have the same structure, they must be equal. 

406 if lhs == rhs: 

407 return S.true # e.g. True == True 

408 elif all(isinstance(i, BooleanAtom) for i in (rhs, lhs)): 

409 return S.false # True != False 

410 elif not (lhs.is_Symbol or rhs.is_Symbol) and ( 

411 isinstance(lhs, Boolean) != 

412 isinstance(rhs, Boolean)): 

413 return S.false # only Booleans can equal Booleans 

414 

415 # check finiteness 

416 fin = L, R = [i.is_finite for i in (lhs, rhs)] 

417 if None not in fin: 

418 if L != R: 

419 return S.false 

420 if L is False: 

421 if lhs == -rhs: # Eq(oo, -oo) 

422 return S.false 

423 return S.true 

424 elif None in fin and False in fin: 

425 return Relational.__new__(cls, lhs, rhs, **options) 

426 

427 if all(isinstance(i, Expr) for i in (lhs, rhs)): 

428 # see if the difference evaluates 

429 dif = lhs - rhs 

430 z = dif.is_zero 

431 if z is not None: 

432 if z is False and dif.is_commutative: # issue 10728 

433 return S.false 

434 if z: 

435 return S.true 

436 # evaluate numerically if possible 

437 n2 = _n2(lhs, rhs) 

438 if n2 is not None: 

439 return _sympify(n2 == 0) 

440 # see if the ratio evaluates 

441 n, d = dif.as_numer_denom() 

442 rv = None 

443 if n.is_zero: 

444 rv = d.is_nonzero 

445 elif n.is_finite: 

446 if d.is_infinite: 

447 rv = S.true 

448 elif n.is_zero is False: 

449 rv = d.is_infinite 

450 if rv is None: 

451 # if the condition that makes the denominator 

452 # infinite does not make the original expression 

453 # True then False can be returned 

454 l, r = clear_coefficients(d, S.Infinity) 

455 args = [_.subs(l, r) for _ in (lhs, rhs)] 

456 if args != [lhs, rhs]: 

457 rv = fuzzy_bool(Eq(*args)) 

458 if rv is True: 

459 rv = None 

460 elif any(a.is_infinite for a in Add.make_args(n)): 

461 # (inf or nan)/x != 0 

462 rv = S.false 

463 if rv is not None: 

464 return _sympify(rv) 

465 

466 return Relational.__new__(cls, lhs, rhs, **options)

694 mathematical notation:

699 inequalities: 

700 

701 +-----------------+--------+ 

702 |Class Name | Symbol | 

703 +=================+========+ 

704 |GreaterThan | (>=) | 

705 +-----------------+--------+ 

706 |LessThan | (<=) | 

707 +-----------------+--------+ 

708 |StrictGreaterThan| (>) | 

709 +-----------------+--------+ 

710 |StrictLessThan | (<) | 

711 +-----------------+--------+ 

712 

713 All classes take two arguments, lhs and rhs. 

714 

715 +----------------------------+-----------------+ 

716 |Signature Example | Math equivalent | 

717 +============================+=================+ 

718 |GreaterThan(lhs, rhs) | lhs >= rhs | 

719 +----------------------------+-----------------+

727 In addition to the normal .lhs and .rhs of Relations, ``*Than`` inequality 

728 objects also have the .lts and .gts properties, which represent the "less 

729 than side" and "greater than side" of the operator. Use of .lts and .gts 

730 in an algorithm rather than .lhs and .rhs as an assumption of inequality 

731 direction will make more explicit the intent of a certain section of code, 

732 and will make it similarly more robust to client code changes: 

733 

734 >>> from sympy import GreaterThan, StrictGreaterThan 

735 >>> from sympy import LessThan, StrictLessThan 

736 >>> from sympy import And, Ge, Gt, Le, Lt, Rel, S 

737 >>> from sympy.abc import x, y, z 

738 >>> from sympy.core.relational import Relational 

739 

740 >>> e = GreaterThan(x, 1) 

741 >>> e 

742 x >= 1 

743 >>> '%s >= %s is the same as %s <= %s' % (e.gts, e.lts, e.lts, e.gts) 

744 'x >= 1 is the same as 1 <= x'

759 Another option is to use the Python inequality operators (>=, >, <=, <)

761 is that one can write a more "mathematical looking" statement rather than 

762 littering the math with oddball function calls. However there are certain 

763 (minor) caveats of which to be aware (search for 'gotcha', below).

765 >>> x >= 2

767 >>> _ == Ge(x, 2) 

768 True

771 succinctly and less conveniently:

773 >>> Rel(x, 1, ">") 

774 x > 1 

775 >>> Relational(x, 1, ">") 

776 x > 1

779 x > 1 

780 >>> GreaterThan(x, 1) 

781 x >= 1 

782 >>> LessThan(x, 1)

785 x < 1

787 Notes 

788 ===== 

789 

790 There are a couple of "gotchas" to be aware of when using Python's 

791 operators. 

792 

793 The first is that what your write is not always what you get: 

794 

795 >>> 1 < x 

796 x > 1 

797 

798 Due to the order that Python parses a statement, it may 

799 not immediately find two objects comparable. When "1 < x" 

800 is evaluated, Python recognizes that the number 1 is a native 

801 number and that x is *not*. Because a native Python number does 

802 not know how to compare itself with a SymPy object 

803 Python will try the reflective operation, "x > 1" and that is the 

804 form that gets evaluated, hence returned. 

805 

806 If the order of the statement is important (for visual output to 

807 the console, perhaps), one can work around this annoyance in a 

808 couple ways: 

809 

810 (1) "sympify" the literal before comparison 

811 

812 >>> S(1) < x 

813 1 < x 

814 

815 (2) use one of the wrappers or less succinct methods described 

816 above 

817 

818 >>> Lt(1, x) 

819 1 < x 

820 >>> Relational(1, x, "<") 

821 1 < x 

822 

823 The second gotcha involves writing equality tests between relationals 

824 when one or both sides of the test involve a literal relational: 

825 

826 >>> e = x < 1; e 

827 x < 1 

828 >>> e == e # neither side is a literal 

829 True 

830 >>> e == x < 1 # expecting True, too 

831 False 

832 >>> e != x < 1 # expecting False 

833 x < 1 

834 >>> x < 1 != x < 1 # expecting False or the same thing as before 

835 Traceback (most recent call last): 

836 ... 

837 TypeError: cannot determine truth value of Relational 

838 

839 The solution for this case is to wrap literal relationals in 

840 parentheses: 

841 

842 >>> e == (x < 1) 

843 True 

844 >>> e != (x < 1) 

845 False 

846 >>> (x < 1) != (x < 1) 

847 False 

848 

849 The third gotcha involves chained inequalities not involving 

850 '==' or '!='. Occasionally, one may be tempted to write: 

851 

852 >>> e = x < y < z 

853 Traceback (most recent call last): 

854 ... 

855 TypeError: symbolic boolean expression has no truth value. 

856 

857 Due to an implementation detail or decision of Python [1]_, 

858 there is no way for SymPy to create a chained inequality with 

859 that syntax so one must use And: 

860 

861 >>> e = And(x < y, y < z) 

862 >>> type( e ) 

863 And 

864 >>> e 

865 (x < y) & (y < z) 

866 

867 Although this can also be done with the '&' operator, it cannot 

868 be done with the 'and' operarator: 

869 

870 >>> (x < y) & (y < z) 

871 (x < y) & (y < z) 

872 >>> (x < y) and (y < z) 

873 Traceback (most recent call last): 

874 ... 

875 TypeError: cannot determine truth value of Relational 

876 

877 .. [1] This implementation detail is that Python provides no reliable 

878 method to determine that a chained inequality is being built. 

879 Chained comparison operators are evaluated pairwise, using "and" 

880 logic (see 

881 http://docs.python.org/2/reference/expressions.html#notin). This 

882 is done in an efficient way, so that each object being compared 

883 is only evaluated once and the comparison can short-circuit. For 

884 example, ``1 > 2 > 3`` is evaluated by Python as ``(1 > 2) and (2 

885 > 3)``. The ``and`` operator coerces each side into a bool, 

886 returning the object itself when it short-circuits. The bool of 

887 the --Than operators will raise TypeError on purpose, because 

888 SymPy cannot determine the mathematical ordering of symbolic 

889 expressions. Thus, if we were to compute ``x > y > z``, with 

890 ``x``, ``y``, and ``z`` being Symbols, Python converts the 

891 statement (roughly) into these steps: 

892 

893 (1) x > y > z 

894 (2) (x > y) and (y > z) 

895 (3) (GreaterThanObject) and (y > z) 

896 (4) (GreaterThanObject.__nonzero__()) and (y > z) 

897 (5) TypeError 

898 

899 Because of the "and" added at step 2, the statement gets turned into a 

900 weak ternary statement, and the first object's __nonzero__ method will 

901 raise TypeError. Thus, creating a chained inequality is not possible. 

902 

903 In Python, there is no way to override the ``and`` operator, or to 

904 control how it short circuits, so it is impossible to make something 

905 like ``x > y > z`` work. There was a PEP to change this, 

906 :pep:`335`, but it was officially closed in March, 2012. 

907
