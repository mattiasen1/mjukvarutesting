8def dispatch(*types, **kwargs): 

9 """ Dispatch function on the types of the inputs 

10 

11 Supports dispatch on all non-keyword arguments. 

12 

13 Collects implementations based on the function name. Ignores namespaces. 

14 

15 If ambiguous type signatures occur a warning is raised when the function is 

16 defined suggesting the additional method to break the ambiguity. 

17 

18 Examples 

19 -------- 

20 

21 >>> from sympy.multipledispatch import dispatch 

22 >>> @dispatch(int) 

23 ... def f(x): 

24 ... return x + 1 

25 

26 >>> @dispatch(float) 

27 ... def f(x): 

28 ... return x - 1 

29 

30 >>> f(3) 

31 4 

32 >>> f(3.0) 

33 2.0 

34 

35 Specify an isolated namespace with the namespace keyword argument 

36 

37 >>> my_namespace = dict() 

38 >>> @dispatch(int, namespace=my_namespace) 

39 ... def foo(x): 

40 ... return x + 1 

41 

42 Dispatch on instance methods within classes 

43 

44 >>> class MyClass(object): 

45 ... @dispatch(list) 

46 ... def __init__(self, data): 

47 ... self.data = data 

48 ... @dispatch(int) 

49 ... def __init__(self, datum): 

50 ... self.data = [datum] 

51 """ 

52 namespace = kwargs.get('namespace', global_namespace) 

53 on_ambiguity = kwargs.get('on_ambiguity', ambiguity_warn) 

54 

55 types = tuple(types) 

56 

57 def _(func): 

58 name = func.__name__ 

59 

60 if ismethod(func): 

61 dispatcher = inspect.currentframe().f_back.f_locals.get( 

62 name, 

63 MethodDispatcher(name)) 

64 else: 

65 if name not in namespace: 

66 namespace[name] = Dispatcher(name) 

67 dispatcher = namespace[name] 

68 

69 dispatcher.add(types, func, on_ambiguity=on_ambiguity) 

70 return dispatcher

74def ismethod(func): 

75 """ Is func a method? 

76 

77 Note that this has to work as the method is defined but before the class is 

78 defined. At this stage methods look like functions. 

79 """ 

80 if hasattr(inspect, "signature"): 

81 signature = inspect.signature(func) 

82 return signature.parameters.get('self', None) is not None 

83 else: 

84 spec = inspect.getargspec(func) 

85 return spec and spec.args and spec.args[0] == 'self'
